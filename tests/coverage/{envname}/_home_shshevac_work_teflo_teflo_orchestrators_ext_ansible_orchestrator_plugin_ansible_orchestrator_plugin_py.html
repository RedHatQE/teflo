37class AnsibleOrchestratorPlugin(OrchestratorPlugin): 

38 """Ansible orchestrator Plugin. 

39 

40 This class primary responsibility is for processing teflo actions. 

41 These actions for the ansible orchestrator plugin could be in the form of a 

42 playbook or module call. 

43 """ 

44 __plugin_name__ = 'ansible' 

45 __schema_file_path__ = os.path.abspath(os.path.join(os.path.dirname(__file__), "files/schema.yml")) 

46 __schema_ext_path__ = os.path.abspath(os.path.join(os.path.dirname(__file__), "files/extensions.py"))

48 def __init__(self, package): 

49 """Constructor. 

50 

51 :param package: action resource 

52 :type package: object 

53 """ 

54 super(AnsibleOrchestratorPlugin, self).__init__(package) 

55 

56 self.options = getattr(package, 'ansible_options', None) 

57 self.galaxy_options = getattr(package, 'ansible_galaxy_options', None) 

58 self.playbook = getattr(package, 'ansible_playbook', None) 

59 self.script = getattr(package, 'ansible_script', None) 

60 self.shell = getattr(package, 'ansible_shell', None) 

61 self.all_hosts = getattr(package, 'all_hosts', [])

63 # calling the method to do a backward compatibility check in case user is defining name field as a path for 

64 # script or playbook 

65 # TODO delete this if we want to remove backward compatibility for later releases 

66 self.backwards_compat_check()

68 # ansible service object 

69 self.ans_service = AnsibleService(self.config, self.hosts, self.all_hosts, 

70 self.options, self.galaxy_options, 

71 concurrency=self.config['TASK_CONCURRENCY']['ORCHESTRATE'].lower())

73 def backwards_compat_check(self): 

74 """ This method does the check if name field is a script/playbook path or name of the orchestrator task by 

75 checking is '/' i spresent in the string. 

76 If it is a path then it checks if ansible_script field is a boolean and is True . If so 

77 a new dictionary is created with key=name and the value= script path. This is then assigned to 

78 ansible_script. 

79 If the ansible_script is not present then it is understood that the path belongs to a playbook. 

80 a new dictionary is created with key=name and the value= playbook path. This is then assigned to 

81 ansible_playbook. 

82 """ 

83 

84 if os.sep in self.action_name: 

85 self.logger.warning('Using name field to provide ansible_script/ansible_playbook path') 

86 

87 self.logger.debug('Joining current workspace %s to the ansible_script/playbook path %s' 

88 % (self.workspace, self.action_name)) 

89 new_item = {'name': os.path.join(self.workspace, self.action_name)} 

90 

91 self.logger.debug('Converting ansible_script/playbook path to dictionary %s' % new_item) 

92 if isinstance(self.script, bool) and self.script: 

93 self.script = new_item 

94 elif not self.script: 

95 self.playbook = new_item 

96 else: 

97 raise TefloOrchestratorError('Error in defining the orchestrate name/ansible_script/ansible_playbook' 

98 ' fields')

100 def validate(self): 

101 """Validate that script/playbook path is valid and exists.""" 

102 

103 # schema validation for ansible_orchestrate schema 

104 schema_validator(schema_data=self.build_profile(self.action), schema_files=[self.__schema_file_path__], 

105 schema_ext_files=[self.__schema_ext_path__]) 

106 

107 # verifying when script or playbook is present in the orchestrate task, the name key provides a path that exist 

108 if self.script: 

109 if os.path.exists(self.script.get('name').split(' ', 1)[0]): 

110 self.logger.debug('Found Action resource script %s' % self.script.get('name')) 

111 else: 

112 raise TefloOrchestratorError('Cannot find Action resource script %s' % self.script.get('name')) 

113 elif self.playbook: 

114 if os.path.exists(self.playbook.get('name').split(' ', 1)[0]): 

115 self.logger.debug('Found Action resource playbook %s' % self.playbook.get('name')) 

116 else: 

117 raise TefloOrchestratorError('Cannot find Action resource playbook %s' % self.playbook.get('name'))

119 def __playbook__(self): 

120 self.logger.info('Executing playbook:') 

121 results = self.ans_service.run_playbook(self.playbook) 

122 if results[0] != 0: 

123 raise TefloOrchestratorError('Playbook %s failed to run' % self.playbook['name']) 

124 else: 

125 self.logger.info('Successfully completed playbook : %s' % self.playbook['name']) 

126 

127 def __script__(self): 

128 self.logger.info('Executing script:') 

129 

130 result = self.ans_service.run_script_playbook(self.script) 

131 if result['rc'] != 0: 

132 raise TefloOrchestratorError('Script %s failed. Host=%s rc=%d Error: %s' 

133 % (self.script['name'], result['host'], result['rc'], result['err'])) 

134 else: 

135 self.logger.info('Successfully completed script : %s' % self.script['name']) 

136 

137 def __shell__(self): 

138 self.logger.info('Executing shell command:') 

139 for shell in self.shell: 

140 result = self.ans_service.run_shell_playbook(shell) 

141 if result['rc'] != 0: 

142 raise TefloOrchestratorError('Command %s failed. Host=%s rc=%d Error: %s' 

143 % (shell['command'], result['host'], result['rc'], result['err'])) 

144 else: 

145 self.logger.info('Successfully completed command : %s' % shell['command']) 

146 

147 def run(self): 

148 """Run method for orchestrator. 

149 """ 

150 # Orchestrate supports only one action_types( playbook, script or shell) per task 

151 # if more than one action types are declared then the first action_type found will be executed 

152 

153 flag = 0 

154 res = self.action.status 

155 for item in ['playbook', 'script', 'shell']: 

156 # Orchestrate supports only one action_types( playbook, script or shell) per task 

157 # if more than one action types are declared then the first action_type found will be executed 

158 if getattr(self, item): 

159 flag += 1 

160 # Download ansible roles (if applicable) 

161 if flag == 1: 

162 try: 

163 self.ans_service.download_roles() 

164 except (TefloOrchestratorError, AnsibleServiceError): 

165 if 'retry' in self.galaxy_options and self.galaxy_options['retry']: 

166 self.logger.Info("Download failed. Sleeping 5 seconds and \ 

167 trying again") 

168 time.sleep(5) 

169 self.ans_service.download_roles() 

170 

171 try: 

172 getattr(self, '__%s__' % item)() 

173 # If every script/playbook/shell command within the each orchestrate has passed, 

174 # mark that task as successful 

175 self.logger.debug("Successful completion of orchestrate task %s with return value %s" 

176 % (self.action_name, res)) 

177 res = 0 

178 except (TefloOrchestratorError, AnsibleServiceError, Exception) as e: 

179 res = 1 

180 self.logger.error("Orchestration failed : %s" % e) 

181 break 

182 finally: 

183 # get ansible logs as needed 

184 # providing folder_name as 'ansible_orchestrator' so all ansible logs wil be under this folder 

185 self.ans_service.alog_update(folder_name='ansible_orchestrator') 

186 else: 

187 self.logger.warning('Found more than one action types (ansible_playbook, ansible_script ,' 

188 'ansible_shell )in the orchestrate task, only the first found' 

189 ' action type was executed, the rest are skipped.') 

190 break 

191 return res
