63 def __check_custom_resource(self): 

64 # method to run playbook or scripts as a part of resource validation on local hosts before the scenario 

65 # provisioning is started 

66 

67 # creating ansible service class object 

68 self.ans_service = AnsibleService(self.config, hosts=['localhost'], all_hosts=[], ansible_options=None) 

69 LOG.info("Running validation using user provided playbook/scripts") 

70 for type in ['playbook', 'script']: 

71 if not getattr(self.scenario, 'resource_check').get(type, None): 

72 continue 

73 for item in getattr(self.scenario, 'resource_check').get(type): 

74 status = 0 

75 self.ans_service.options = item.get('ansible_options', None) 

76 self.ans_service.galaxy_options = item.get('ansible_galaxy_options', None) 

77 

78 error_msg = '' 

79 try: 

80 if type == 'script': 

81 # running the script 

82 result = self.ans_service.run_script_playbook(item) 

83 if result['rc'] != 0: 

84 status = 1 

85 LOG.error('Script %s failed with return code %s and error %s' % 

86 (item['name'], result['rc'], result['err'])) 

87 error_msg = result['err'] 

88 else: 

89 # running the playbook 

90 result = self.ans_service.run_playbook(item) 

91 if result[0] != 0: 

92 status = 1 

93 error_msg = result[1] 

94 

95 if status == 1: 

96 raise TefloError("Failed to run validation playbook/script %s. ERROR: %s" % 

97 (item['name'], error_msg)) 

98 else: 

99 LOG.info("Successfully completed resource_check validation for playbook/script: %s", 

100 item['name']) 

101 except TefloError: 

102 raise TefloError("Failed to run resource_check validation for playbook/script. ERROR: %s" 

103 % error_msg)

105 def __check_service(self): 

106 """ 

107 External Component Dependency Check 

108 Throws exception if all components are not UP 

109 

110 :param scenario: teflo scenario object 

111 :param config: teflo config object 

112 """ 

113 

114 # External Dependency Check 

115 # Available components to check ci-rhos, zabbix-sysops, brew, covscan 

116 # polarion, rpmdiff, umb, errata, rdo-cloud 

117 # gerrit 

118 

119 # Verify dependency check components are supported/valid then 

120 # Check status (UP/DOWN) 

121 # Only check if dependency check endpoint set and components given 

122 # Else it is ignored 

123 

124 LOG.info('Running external resource validation') 

125 if self.config['RESOURCE_CHECK_ENDPOINT']: 

126 endpoint = self.config['RESOURCE_CHECK_ENDPOINT'] 

127 ext_resources_avail = True 

128 # component_names = self.scenario.resource_check['service'] 

129 component_names = getattr(self.scenario, 'resource_check').get('monitored_services', None) 

130 urllib3.disable_warnings() 

131 components = cachet.Components(endpoint=endpoint, verify=False) 

132 LOG.info(' DEPENDENCY CHECK '.center(64, '-')) 

133 for comp in component_names: 

134 comp_resource_invalid = False 

135 comp_resource_avail = False 

136 for attempts in range(1, 6): 

137 component_data = components.get(params={'name': comp}) 

138 if json.loads(component_data)['data']: 

139 comp_status = json.loads(component_data)['data'][0]['status'] 

140 if comp_status == 4: 

141 comp_resource_avail = False 

142 time.sleep(30) 

143 continue 

144 else: 

145 comp_resource_avail = True 

146 break 

147 else: 

148 comp_resource_invalid = True 

149 if comp_resource_avail is not True or comp_resource_invalid is True: 

150 ext_resources_avail = False 

151 if comp_resource_invalid: 

152 LOG.info('{:>40} {:<9} - Attempts {}'.format( 

153 comp.upper(), ': INVALID', attempts)) 

154 else: 

155 LOG.info('{:>40} {:<9} - Attempts {}'.format( 

156 comp.upper(), ': UP' if comp_resource_avail else ': DOWN', attempts)) 

157 warnings.resetwarnings() 

158 LOG.info(''.center(64, '-')) 

159 

160 if ext_resources_avail is not True: 

161 LOG.error("ERROR: Not all external resources are available or valid. Not running scenario") 

162 raise TefloError('Scenario %s will not be run! Not all external resources are available or valid' % 

163 getattr(self.scenario, 'name'))
