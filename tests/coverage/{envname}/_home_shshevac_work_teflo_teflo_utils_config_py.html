53 def __set_defaults__(self): 

54 """Set the default configuration settings.""" 

55 # A check to continue the flow if default section is not provided in teflo.cfg 

56 if getattr(self.parser, '_sections').get('defaults'): 

57 for k, v in getattr(self.parser, '_sections')['defaults'].items(): 

58 if k == '__name__': 

59 continue 

60 self.__setitem__(k.upper(), v)

62 def __set_credentials__(self, **kwargs): 

63 """Set the credentials configuration settings.""" 

64 if not kwargs.get("parser"): 

65 parser = self.parser 

66 else: 

67 parser = kwargs["parser"] 

68 credentials = [] 

69 

70 for section in getattr(parser, '_sections'): 

71 if not section.startswith('credentials'): 

72 continue 

73 

74 _credentials = {} 

75 

76 for option in parser.options(section): 

77 _credentials[option] = \ 

78 parser.get(section, option) 

79 _credentials['name'] = section.split(':')[-1] 

80 credentials.append(_credentials) 

81 

82 self.__setitem__('CREDENTIALS', credentials)

211 def __set_timeout__(self): 

212 """ 

213 Set timeout for each tasks. The task will be terminated within the time 

214 you set up from teflo.cfg [timrout] section 

215 e.x.: 

216 [timeout] 

217 provision=500 

218 orchestrate=500 

219 cleanup=700 

220 report=100 

221 """ 

222 _timeout = DEFAULT_TIMEOUT 

223 for section in getattr(self.parser, "_sections"): 

224 if not section.startswith("timeout"): 

225 continue 

226 for option in self.parser.options(section): 

227 _timeout.update({option.upper(): int(self.parser.get(section, option))}) 

228 self.__setitem__('TIMEOUT', _timeout)

230 def load(self): 

231 """Load configuration settings. 

232 

233 Configuration will be loaded from the following order: 

234 - /etc/teflo/teflo.cfg 

235 - ./teflo.cfg 

236 - TEFLO_SETTINGS env variable setting the config file 

237 """ 

238 files = [ 

239 '/etc/teflo/teflo.cfg', 

240 os.path.join(os.getcwd(), 'teflo.cfg') 

241 ] 

242 

243 if os.getenv('TEFLO_SETTINGS'): 

244 files.append(os.getenv('TEFLO_SETTINGS')) 

245 

246 for filename in files: 

247 if not os.path.exists(filename): 

248 # file not found 

249 continue 

250 

251 # read the file 

252 self.parser.read(filename) 

253 

254 # set user supplied configuration settings overriding defaults 

255 for config in DEFAULT_CONFIG_SECTIONS: 

256 getattr(self, '__set_%s__' % config)() 

257 

258 cred_man = AnsibleCredentialManager(self) 

259 cred_man.populate_teflo_cfg_credentials()
