41class Action(TefloResource): 

42 """ 

43 The action resource class. The teflo compound can contain x amount of 

44 actions. Their primary responsibility is to do some sort of work. Actions 

45 can handle any sort of task to be processed. Most of the time you will 

46 see actions consisting of the following: 

47 - System configuration 

48 - Product installation/configuration 

49 - Test setup (test framework installation/configuration) 

50 

51 Each action has an associated orchestrator. The orchestrator is what 

52 processes (completes) the given action. 

53 """ 

54 

55 _valid_tasks_types = ['validate', 'orchestrate', 'cleanup'] 

56 _fields = ['name', 'description', 'orchestrator', 'hosts', 'labels', 'status' 'orchestrate_timeout', 

57 'cleanup_timeout', 'validate_timeout']

59 def __init__(self, 

60 config=None, 

61 name=None, 

62 parameters=dict, 

63 validate_task_cls=ValidateTask, 

64 orchestrate_task_cls=OrchestrateTask, 

65 cleanup_task_cls=CleanupTask, 

66 **kwargs): 

67 """Constructor. 

68 

69 The primary focus of the constructor is to build the action object 

70 containing all necessary parts to process the action. 

71 

72 :param config: teflo configuration 

73 :type config: dict 

74 :param name: action resource name 

75 :type name: str 

76 :param parameters: content which makes up the action resource 

77 :type parameters: dict 

78 :param validate_task_cls: teflos validate task class 

79 :type validate_task_cls: object 

80 :param orchestrate_task_cls: teflos orchestrate task class 

81 :type orchestrate_task_cls: object 

82 :param cleanup_task_cls: teflos cleanup task class 

83 :type cleanup_task_cls: object 

84 :param kwargs: additional key:value(s) 

85 :type kwargs: dict 

86 """ 

87 super(Action, self).__init__(config=config, name=name, **kwargs) 

88 

89 # set the action resource name 

90 if name is None: 

91 self._name = parameters.pop('name', None) 

92 if self._name is None: 

93 raise TefloActionError('Unable to build action object. Name' 

94 ' field missing!') 

95 else: 

96 self._name = name 

97 

98 # set the timeout for VALIDATE 

99 try: 

100 if parameters.get('validate_timeout') is not None: 

101 self._validate_timeout = parameters.pop("validate_timeout") 

102 else: 

103 self._validate_timeout = config["TIMEOUT"]["VALIDATE"] 

104 except TypeError: 

105 self.logger.error("No teflo.cfg found, so no timeout will be set") 

106 self._validate_timeout = 0 

107 

108 # set the timeout for ORCHESTRATE 

109 try: 

110 if parameters.get('orchestrate_timeout') is not None: 

111 self._orchestrate_timeout = parameters.pop("orchestrate_timeout") 

112 else: 

113 self._orchestrate_timeout = config["TIMEOUT"]["ORCHESTRATE"] 

114 except TypeError: 

115 self.logger.error("No teflo.cfg found, so no timeout will be set") 

116 self._orchestrate_timeout = 0 

117 

118 # set the timeout for cleanup 

119 try: 

120 if parameters.get('cleanup_timeout') is not None: 

121 self._cleanup_timeout = parameters.pop("cleanup_timeout") 

122 else: 

123 self._cleanup_timeout = config["TIMEOUT"]["CLEANUP"] 

124 except TypeError: 

125 self.logger.error("No teflo.cfg found, so no timeout will be set") 

126 self._cleanup_timeout = 0 

127 

128 # set the action description 

129 self._description = parameters.pop('description', None) 

130 

131 # each action will have x number of hosts associated to it. lets 

132 # associate the list of hosts to the action object itself. currently 

133 # the hosts are strings, when teflo builds the pipeline, they will 

134 # be updated with their corresponding host object. 

135 self.hosts = parameters.pop('hosts') 

136 if self.hosts is None: 

137 raise TefloActionError('Unable to associate hosts to action: %s.' 

138 'No hosts defined!' % self._name) 

139 

140 # convert the hosts into list format if hosts defined as str format 

141 if isinstance(self.hosts, string_types): 

142 self.hosts = self.hosts.replace(' ', '').split(',') 

143 

144 # every action has a mandatory orchestrator, lets set it 

145 orchestrator_name = parameters.pop('orchestrator', ORCHESTRATOR) 

146 

147 if orchestrator_name not in get_orchestrators_plugin_list(): 

148 raise TefloActionError('Orchestrator: %s is not supported!' % 

149 orchestrator_name) 

150 

151 # now that we know the orchestrator, lets get the class 

152 self._orchestrator = get_orchestrator_plugin_class(orchestrator_name) 

153 

154 # using plugin's method to get the schema keys and check if they are present in the parameters and then 

155 # set them as action's parameters 

156 if self.orchestrator: 

157 for p in getattr(self.orchestrator, 'get_schema_keys')(): 

158 if parameters.get(p, None): 

159 setattr(self, p, parameters.get(p)) 

160 else: 

161 raise TefloActionError('Orchestrator: %s plugin was not found!' % 

162 orchestrator_name) 

163 

164 # set up status code 

165 self._status = parameters.pop('status', 0) 

166 

167 # set labels 

168 setattr(self, 'labels', parameters.pop('labels', [])) 

169 

170 # ******** CLEANUP ACTIONS ******** # 

171 # check if the action requires any cleanup actions prior to host 

172 # deletion 

173 self.cleanup = None 

174 self.cleanup_def = parameters.pop('cleanup', None) 

175 

176 if self.cleanup_def: 

177 # create the cleanup sub-action for this parent action 

178 self.cleanup = Action( 

179 config=self.config, 

180 parameters=copy(self.cleanup_def) 

181 ) 

182 

183 # set the teflo task classes for the resource 

184 self._validate_task_cls = validate_task_cls 

185 self._orchestrate_task_cls = orchestrate_task_cls 

186 self._cleanup_task_cls = cleanup_task_cls 

187 

188 # reload construct task methods 

189 self.reload_tasks() 

190 

191 # load the parameters set into the object itself 

192 if parameters: 

193 self.load(parameters)

217 def profile(self): 

218 """Builds a profile for the action resource. 

219 

220 :return: the action profile 

221 :rtype: OrderedDict 

222 """ 

223 profile = OrderedDict() 

224 profile['name'] = self.name 

225 profile['description'] = self.description 

226 if isinstance(self.orchestrator, string_types): 

227 profile['orchestrator'] = self.orchestrator 

228 else: 

229 profile['orchestrator'] = getattr( 

230 self.orchestrator, '__plugin_name__') 

231 # 

232 # set the action's hosts 

233 if all(isinstance(item, string_types) for item in self.hosts): 

234 profile.update(hosts=[host for host in self.hosts]) 

235 else: 

236 profile.update(dict(hosts=[host.name for host in self.hosts])) 

237 

238 # Update profile with all the parameters for the orchestrator 

239 if not isinstance(self.orchestrator, string_types): 

240 profile.update(getattr(self.orchestrator, 'build_profile')(self)) 

241 

242 if self.cleanup_def: 

243 profile.update({'cleanup': self.cleanup_def}) 

244 

245 # set labels 

246 profile.update({'labels': self.labels}) 

247 profile.update({'status': self.status}) 

248 

249 return profile
