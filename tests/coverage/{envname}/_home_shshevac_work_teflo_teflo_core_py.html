45class LoggerMixin(object): 

46 """Teflos logger mixin class. 

47 

48 This class provides an easy interface for other classes throughout teflo 

49 to utilize the teflo logger. 

50 

51 When a teflo object is created, the teflo logger will be created also. 

52 Allowing easy access to the logger as follows: 

53 

54 cbn = Teflo() 

55 cbn.logger.info('Teflo!') 

56 

57 Teflo packages (classes) can either include the logger mixin or create 

58 their own object. 

59 

60 class Asset(TefloResource): 

61 self.logger.info('Teflo Resource!') 

62 

63 Modules that want to use teflo logger per function base and not per class, 

64 can access the teflo logger as follows: 

65 

66 from logging import getLogger 

67 LOG = getLogger(__name__) 

68 LOG.info('Teflo!') 

69 

70 New loggers for other libraries can easily be added. A create_lib_logger 

71 method will need to be create to setup the logger. Then lastly you can set 

72 a property to return that specific logger for easy access. 

73 """ 

74 

75 _DEBUG_LOG_FORMAT = ("%(asctime)s %(levelname)s [%(name)s.%(funcName)s:%(lineno)d] %(message)s") 

76 _INFO_LOG_FORMAT = ("%(asctime)s %(levelname)s %(message)s") 

77 

78 _LOG_LEVELS = { 

79 'debug': DEBUG, 

80 'info': INFO, 

81 'warning': WARNING, 

82 'error': ERROR, 

83 'critical': CRITICAL 

84 } 

85 

86 @classmethod 

87 def setup_logger(cls, name, file_path, config=None): 

88 

89 # Setting up handlers 

90 LOGGING_CONFIG['handlers']['file'].update({'filename': file_path}) 

91 

92 # setup logging formatters 

93 LOGGING_CONFIG['formatters']['default'].update({'format': cls._INFO_LOG_FORMAT}) 

94 LOGGING_CONFIG['formatters']['debug'].update({'format': cls._DEBUG_LOG_FORMAT}) 

95 

96 if config: 

97 # setup any other loggers that might have been specified in the teflo.cfg 

98 # For now this might suffice but if the teflo community changes we can look 

99 # at a better way through the plugins 

100 for logger in config['SETUP_LOGGER']: 

101 LOGGING_CONFIG['loggers'].update({logger: {'handlers': ['console', 'file'], 

102 'level': cls._LOG_LEVELS[config['LOG_LEVEL']], 

103 'propagate': False}}) 

104 log_level = config['LOG_LEVEL'] 

105 else: 

106 log_level = 'info' if getLogger('teflo').getEffectiveLevel() == 20 else 'debug' 

107 

108 if log_level == 'debug': 

109 

110 for handler in LOGGING_CONFIG['handlers']: 

111 LOGGING_CONFIG['handlers'][handler].update({'formatter': 'debug'}) 

112 LOGGING_CONFIG['handlers'][handler].update({'level': cls._LOG_LEVELS[log_level]}) 

113 

114 for logger in LOGGING_CONFIG['loggers']: 

115 LOGGING_CONFIG['loggers'][logger].update({'level': cls._LOG_LEVELS[log_level]}) 

116 

117 # Configure the individual logger, name is the logger name provided during creation 

118 LOGGING_CONFIG['loggers'].update({name: {'handlers': ['console', 'file'], 

119 'level': cls._LOG_LEVELS[log_level], 

120 'propagate': False}}) 

121 

122 # Init the logging config 

123 log_config.dictConfig(LOGGING_CONFIG) 

124 

125 @classmethod 

126 def create_logger(cls, name, config=None, data_folder=None): 

127 """Create logger. 

128 

129 This method will create logger's to be used throughout teflo. 

130 

131 :param data_folder: name of teflo's data folder 

132 :type data_folder: str 

133 :param name: Name for the logger to create. 

134 :type name: str 

135 :param config: Teflo config object. 

136 :type config: dict 

137 

138 """ 

139 

140 # create log directory 

141 if config: 

142 log_dir = os.path.join(config['DATA_FOLDER'], 'logs') 

143 else: 

144 log_dir = os.path.join(data_folder, 'logs') 

145 

146 try: 

147 

148 if not os.path.exists(log_dir): 

149 os.makedirs(log_dir) 

150 except OSError as ex: 

151 msg = 'Unable to create %s directory' % log_dir 

152 if ex.errno == errno.EACCES: 

153 msg += ', permission defined.' 

154 else: 

155 msg += ', %s.' % ex 

156 raise LoggerMixinError(msg) 

157 

158 full_path = os.path.join(log_dir, 'teflo_scenario.log') 

159 

160 # setup and initialize LOGGING_CONFIG 

161 cls.setup_logger(name, full_path, config) 

162

165 """Returns the default logger (teflo logger) object.""" 

166 return getLogger(inspect.getmodule(inspect.stack()[1][0]).__name__)

177class TimeMixin(object): 

178 """Teflo's time mixin class. 

179 

180 This class provides an easy interface for other teflo classes to save 

181 a start and end time. Once times are saved they can calculate the time 

182 delta between the two points in time. 

183 """ 

184 _start_time = None 

185 _end_time = None 

186 _hours = 0 

187 _minutes = 0 

188 _secounds = 0 

189 

190 def start(self): 

191 """Set the start time.""" 

192 self._start_time = time() 

193 

194 def end(self): 

195 """Set the end time.""" 

196 self._end_time = time() 

197 

198 # calculate time delta 

199 delta = self._end_time - self._start_time 

200 self.hours = delta // 3600 

201 delta = delta - 3600 * self.hours 

202 self.minutes = delta // 60 

203 self.seconds = delta - 60 * self.minutes

277class FileLockMixin(object): 

278 """ 

279 The FileLockMixin is designed to 

280 use file locks to be able to read/write 

281 to a file when multipleprocesses need to 

282 access the same file. 

283 """ 

284 _lock_file = '/tmp/cbn.lock' 

285 _lock_sleep = 5 

286 _lock_timeout = 120 

287 

288 @property 

289 def lock_file(self): 

290 return self._lock_file 

291 

292 @lock_file.setter 

293 def lock_file(self, value): 

294 self._lock_file = value 

295 

296 @property 

297 def lock_sleep(self): 

298 return self._lock_sleep 

299 

300 @lock_sleep.setter 

301 def lock_sleep(self, value): 

302 self._lock_sleep = value 

303 

304 @property 

305 def lock_timeout(self): 

306 return self._lock_timeout 

307 

308 @lock_timeout.setter 

309 def lock_timeout(self, value): 

310 self._lock_timeout = value 

311 

312 def acquire(self): 

313 self.cleanup_locks() 

314 if self._check_and_sleep(): 

315 open(self._lock_file, 'w').close() 

316 

317 def release(self): 

318 try: 

319 os.remove(self._lock_file) 

320 except OSError: 

321 raise 

322 

323 def _check_and_sleep(self): 

324 

325 attempts = 0 

326 total_attempts = self.lock_timeout / self.lock_sleep 

327 

328 while os.path.exists(self.lock_file): 

329 if attempts < total_attempts: 

330 sleep(self.lock_sleep) 

331 attempts += 1 

332 else: 

333 raise TefloError('Timed out waiting for the lock to release') 

334 return True 

335 

336 def cleanup_locks(self): 

337 if glob(os.path.join(os.path.dirname(self.lock_file), 'cbn_*')): 

338 for f in glob(os.path.join(os.path.dirname(self.lock_file), 'cbn_*')): 

339 if f != self.lock_file: 

340 os.remove(f)

508 def load(self, data): 

509 """ 

510 Given a dictionary of attributes, this function loads these 

511 attributes for this class. 

512

515 for key, value in data.items(): 

516 # name has precedence when coming from a YAML file

642 def validate_req_params(self, resource): 

643 """Validate the required parameters exists in the host resource. 

644 

645 :param host: host resource 

646 :type host: object 

647 """ 

648 for item in self.req_params: 

649 name = getattr(resource, 'name') 

650 param, param_type = item[0], item[1] 

651 msg = "Resource %s : required param '%s' " % (name, param) 

652 try: 

653 param_value = getattr(resource, 'provider_params')[param] 

654 self.logger.info(msg + 'exists.') 

655 

656 if not type(param_value) in param_type: 

657 self.logger.error( 

658 ' - Type=%s, Required Type=%s. (ERROR)' % 

659 (type(param_value), param_type)) 

660 raise TefloError( 

661 'Error occurred while validating required provider ' 

662 'parameters for resource %s' % getattr(resource, 'name') 

663 ) 

664 except KeyError: 

665 msg = msg + 'does not exist.' 

666 self.logger.error(msg) 

667 raise TefloError(msg)

1469 else: 

1470 self.inv_dir = os.path.expandvars(os.path.expanduser(os.path.join(self.static_inv_dir, 'inventory'))) 

1471 if not os.path.isdir(os.path.abspath(self.inv_dir)): 

1472 os.makedirs(self.inv_dir)

1480 def create_master(self, all_hosts): 

1481 """Create the master ansible inventory. 

1482 This method will create a master inventory which contains all the 

1483 hosts in the given scenario. Each host will have a group/group:vars. 

1484 """ 

1485 try: 

1486 

1487 # get the lock 

1488 self.acquire() 

1489 

1490 # check for any old master inventories and delete them. 

1491 # Check both the static inv folder as well as the .results/inventory folder for old master inventories 

1492

1494 stat_inv_dir = os.path.join(self.inv_dir, 'master*')
