67def get_core_tasks_classes(): 

68 """ 

69 Go through all modules within teflo.tasks package and return 

70 the list of all tasks classes within it. All tasks within the teflo.tasks 

71 module are considered valid task class to be added into the pipeline. 

72 :return: List of all valid tasks classes 

73 """ 

74 from .core import TefloTask 

75 from . import tasks 

76 

77 # all task classes must 

78 prefix = tasks.__name__ + "." 

79 

80 tasks_list = [] 

81 

82 # Run through each module within tasks and take the list of 

83 # classes that are subclass of TefloTask but not TefloTask itself. 

84 # When you import a class within a module, it becames a member of 

85 # that class 

86 for importer, modname, ispkg in pkgutil.iter_modules(tasks.__path__, prefix): 

87 if str(modname).endswith('.ext'): 

88 continue 

89 clsmembers = inspect.getmembers(sys.modules[modname], inspect.isclass) 

90 for clsname, clsmember in clsmembers: 

91 if (clsmember is not TefloTask) and issubclass(clsmember, TefloTask): 

92 tasks_list.append(clsmember) 

93 

94 return tasks_list

348def schema_validator(schema_data, schema_files, schema_creds=None, schema_ext_files=None): 

349 """ 

350 

351 :param schema_data: the schema dictionary data 

352 :type dict 

353 :param schema_files: the yaml schema file for the plugins 

354 :type list of file paths 

355 :param schema_creds: optional dictionary creds 

356 :type dict 

357 :param schema_ext_files: optional list of extension file paths 

358 :type: list of file paths 

359 :return: 

360 """ 

361 

362 schema = {} 

363 

364 if schema_creds: 

365 schema = {k: v for k, v in schema_creds.items() if k != 'name'} 

366 schema.update({k: v for k, v in schema_data.items() if k != 'credential'}) 

367 else: 

368 schema.update({k: v for k, v in schema_data.items() if k != 'credential'}) 

369 creds = {k: v for k, v in schema_data.items() if k == 'credential'} 

370 if creds: 

371 creds = dict( 

372 credential={x: y for k, v in creds.items() for x, y in v.items() if x != 'name'}) 

373 schema.update(creds) 

374 

375 c = Core(source_data=schema, 

376 schema_files=schema_files, 

377 extensions=schema_ext_files) 

378 

379 try: 

380 c.validate(raise_exception=True) 

381 except (CoreError, SchemaError) as ex: 

382 LOG.error(ex.msg) 

383 raise

399def file_mgmt(operation, file_path, content=None, cfg_parser=None): 

400 """A generic function to manage files (read/write). 

401 

402 :param operation: File operation type to perform 

403 :type operation: str 

404 :param file_path: File name including path 

405 :type file_path: str 

406 :param content: Data to write to a file 

407 :type content: object 

408 :param cfg_parser: Config parser object (Only needed if the file being 

409 processed is a configuration file parser language) 

410 :type cfg_parser: bool 

411 :return: Data that was read from a file 

412 """ 

413 # to maintain the sequence in the results.yml file with ruamel 

414 yaml = YAML() 

415 yaml.default_flow_style = False 

416 yaml.representer.ignore_aliases = lambda *data: True 

417 yaml.Representer.add_representer(OrderedDict, yaml.Representer.represent_dict) 

418 

419 # Determine file extension 

420 file_ext = os.path.splitext(file_path)[-1] 

421 

422 if operation in ['r', 'read']: 

423 # Read 

424 if os.path.exists(file_path): 

425 if file_ext == ".json": 

426 # json 

427 with open(file_path) as f_raw: 

428 return json.load(f_raw) 

429 elif file_ext in ['.yaml', '.yml']: 

430 # yaml 

431 with open(file_path) as f_raw: 

432 return yaml.load(f_raw) 

433 else: 

434 # text 

435 with open(file_path) as f_raw: 

436 if cfg_parser is not None: 

437 # Config parser file 

438 return cfg_parser.readfp(f_raw) 

439 else: 

440 # lets check if it is json 

441 data = f_raw.read() 

442 try: 

443 return json.load(data) 

444 except Exception: 

445 # it wasn't json, lets try yaml 

446 try: 

447 return yaml.load(data) 

448 except Exception: 

449 # it wasn't yaml, lets just return pure string 

450 return data 

451 else: 

452 raise IOError("%s file not found!" % file_path) 

453 elif operation in ['w', 'write']: 

454 # Write 

455 mode = 'w+' if os.path.exists(file_path) else 'w' 

456 if file_ext == ".json": 

457 # json 

458 with open(file_path, mode) as f_raw: 

459 json.dump(content, f_raw, indent=4, sort_keys=True) 

460 elif file_ext in ['.yaml', '.yml']: 

461 # yaml 

462 with open(file_path, mode) as f_raw: 

463 yaml.dump(content, f_raw) 

464 else: 

465 # text 

466 with open(file_path, mode) as f_raw: 

467 if cfg_parser is not None: 

468 # Config parser file 

469 cfg_parser.write(f_raw) 

470 else: 

471 f_raw.write(content) 

472 elif operation in ['d', 'delete']: 

473 if os.path.exists(file_path): 

474 os.unlink(file_path) 

475 else: 

476 raise HelpersError("Unknown file operation: %s." % operation)

531def exec_local_cmd_pipe(cmd, logger, env_var=None): 

532 """Execute command locally, and pipe output in real time. 

533 

534 :param cmd: command to run 

535 :type cmd: str 

536 :param env_var: a dictionary of environmental variables to pass to the subprocess 

537 :type env_var: dictionary 

538 :param logger: logger object 

539 :type logger: object 

540 :return: tuple of rc and error (if there was an error) 

541 """ 

542 # updating passed env variables with os env variables 

543 if env_var: 

544 env_var.update(os.environ) 

545 proc = subprocess.Popen( 

546 cmd, 

547 shell=True, 

548 stdout=subprocess.PIPE, 

549 stderr=subprocess.PIPE, 

550 bufsize=2, 

551 close_fds=True, 

552 env=env_var 

553 ) 

554 while True: 

555 output, error = ("", "") 

556 if proc.poll is not None: 

557 output = proc.stdout.readline().decode('utf-8') 

558 if output == "" and error == "" and proc.poll() is not None: 

559 break 

560 if output: 

561 logger.info(output.strip()) 

562 rc = proc.poll() 

563 if rc != 0: 

564 error = proc.stderr.readline().decode('utf-8') 

565 return rc, error

568class CustomDict(dict):

570 

571 Initialized a data (loaded JSON or YAML) and creates a

576 are also represented as Resource 

577 """ 

578 

579 def __init__(self, data={}): 

580 super(CustomDict, self).__init__(data) 

581 for key, value in data.items(): 

582 if isinstance(value, dict): 

583 value = CustomDict(value) 

584 self[key] = value 

585 

586 def __getattr__(self, attr): 

587 return self.get(attr) 

588 

589 def __setattr__(self, key, value): 

590 self.__setitem__(key, value)

620def filter_notifications_on_trigger(state, notify_list, passed_tasks, failed_tasks): 

621 """ 

622 Go through the notify_list and return notifications with which were skipped. 

623 If no notification with on_demand is found the original list is returned 

624 :return: List of notifications based on_demand 

625 """ 

626 if state == 'on_demand': 

627 return [res for res in notify_list if getattr(res, 'on_demand')] 

628 else: 

629 # filter out all on_demand notifications 

630 notify_list = [res for res in notify_list if not getattr(res, 'on_demand')] 

631 

632 if state == 'on_start': 

633 return [res for res in notify_list if 

634 getattr(res, 'on_start') and len(set(getattr(res, 'on_tasks', [])). 

635 intersection(passed_tasks)) != 0] 

636 elif state == 'on_complete': 

637 ocl = list() 

638 

639 # filter out all on_start notifications 

640 nl = [res for res in notify_list if not getattr(res, 'on_start')] 

641 

642 # not executing if on_success but there are failed tasks 

643 passed = [nt for nt in nl if 

644 getattr(nt, 'on_success') is True and getattr(nt, 'on_failure') is False] 

645 passed = [nt for nt in passed if 

646 len(failed_tasks) == 0 and len(set(getattr(nt, 'on_tasks')). 

647 intersection(passed_tasks)) != 0] 

648 ocl.extend(passed) 

649 

650 # not executing if on_failure but there are passed tasks 

651 failed = [nt for nt in nl if 

652 getattr(nt, 'on_failure') is True and getattr(nt, 'on_success') is False] 

653 failed = [nt for nt in failed if 

654 len(passed_tasks) == 0 and len(set(getattr(nt, 'on_tasks')). 

655 intersection(failed_tasks)) != 0] 

656 ocl.extend(failed) 

657 

658 # not executing if on_failure or on_success 

659 mixed = [nt for nt in nl if 

660 getattr(nt, 'on_success') is True and getattr(nt, 'on_failure') is True] 

661 mixed = [nt for nt in mixed if len(set(getattr(nt, 'on_tasks')). 

662 intersection(failed_tasks)) != 0 or len( 

663 set(getattr(nt, 'on_tasks')). 

664 intersection(passed_tasks)) != 0] 

665 ocl.extend(mixed) 

666 

667 return ocl

832def ssh_retry(obj): 

833 """ 

834 Decorator to check SSH Connection before method execution. 

835 Will perform 30 retries with sleep of 10 seconds 

836 between attempts 

837 """ 

838 MAX_ATTEMPTS = 30 

839 MAX_WAIT_TIME = 10 

840 

841 def check_access(*args, **kwargs): 

842 """ 

843 SSH Connection check and retries 

844 """ 

845 # Set flag and Inventory 

846 ssh_errs = False 

847 args[0].set_inventory() 

848 

849 # put everything into a list for rather than doing repetative if else statements 

850 # especially now that the inventory 'groups' property can be a string list of hosts 

851 # in the master inventory vs what is in the unique inventory 

852 host_groups = [kwargs['extra_vars']['hosts']] if kwargs['extra_vars']['hosts'].find( 

853 ', ') == -1 \ 

854 else kwargs['extra_vars']['hosts'].split(', ') 

855 inv_groups = args[0].inventory.groups 

856 for host_group in host_groups: 

857 if is_host_localhost(host_group): 

858 # Run Playbook/Module if localhost; no need to check. 

859 result = obj(*args, **kwargs) 

860 return result 

861 if host_group not in inv_groups: 

862 raise HelpersError( 

863 'ERROR: Unexpected error - Group %s not found in inventory file!' % 

864 kwargs['extra_vars']['hosts'] 

865 ) 

866 

867 def can_connect(group): 

868 

869 sys_vars = group.vars 

870 server_ip = group.hosts[0].address 

871 LOG.info(server_ip) 

872 

873 # skip ssh connectivity check if server is localhost 

874 if is_host_localhost(server_ip): 

875 return False 

876 

877 server_user = sys_vars['ansible_user'] 

878 server_key_file = sys_vars['ansible_ssh_private_key_file'] 

879 server_ssh_port = 22 if 'ansible_port' not in sys_vars else sys_vars.get('ansible_port') 

880 

881 # Perform SSH checks 

882 attempt = 1 

883 while attempt <= MAX_ATTEMPTS: 

884 try: 

885 ssh = SSHClient() 

886 ssh.set_missing_host_key_policy(WarningPolicy()) 

887 

888 # Test ssh connection 

889 ssh.connect(server_ip, 

890 port=server_ssh_port, 

891 username=server_user, 

892 key_filename=server_key_file, 

893 timeout=5) 

894 LOG.debug("Server %s - IP: %s is reachable." % 

895 (group, server_ip)) 

896 ssh.close() 

897 break 

898 except (BadHostKeyException, AuthenticationException, 

899 SSHException, socket.error) as ex: 

900 attempt = attempt + 1 

901 LOG.error(ex) 

902 LOG.error("Server %s - IP: %s is unreachable." % (group, 

903 server_ip)) 

904 if attempt <= MAX_ATTEMPTS: 

905 LOG.info('Attempt %s of %s: retrying in %s seconds' % 

906 (attempt, MAX_ATTEMPTS, MAX_WAIT_TIME)) 

907 time.sleep(MAX_WAIT_TIME) 

908 

909 # Check Max SSH Retries performed 

910 if attempt > MAX_ATTEMPTS: 

911 LOG.error( 

912 'Max Retries exceeded. SSH ERROR - Resource unreachable - Server %s - IP: %s!' % 

913 (group, server_ip) 

914 ) 

915 return True 

916 return False 

917 

918 for host_group in host_groups: 

919 inv_group = inv_groups[host_group] 

920 # This is just here for backwards compat. In case I've missed any 

921 # corner case 

922 if hasattr(inv_group, 'child_groups') and inv_group.child_groups: 

923 LOG.debug('In the child group block') 

924 for group in inv_group.child_groups: 

925 ssh_errs = can_connect(group) 

926 else: 

927 # Most cases should be falling into this block, 

928 # based on teflo returning the actual host asset name once its 

929 # done with its fetch_assets logic 

930 ssh_errs = can_connect(inv_group) 

931 

932 # Check for SSH Errors 

933 if ssh_errs: 

934 raise HelpersError( 

935 'ERROR: Unable to establish ssh connection with resources!' 

936 ) 

937 

938 # Run Playbook/Module 

939 result = obj(*args, **kwargs) 

940 return result 

941 

942 return check_access

967class DataInjector(object): 

968 """Data injector class. 

969 

970 This class is primarily used for injecting data into strings which the 

971 data to be replaced needs to come from a host resource. It is helpful 

972 in the cases where orchestrate or execute tasks require additional data. 

973 i.e. ip address, metadata, authentication details, etc. 

974 

975 --- 

976 

977 How does this work? 

978 

979 You have a command in your definition file that needs the ip address of a 

980 host resource. 

981 

982 command: /usr/bin/foo --ip { host01.ip_address[0] } --args .. 

983 

984 The host01 is a resource defined in the provision section of teflo and 

985 the ip_address is an attribute of the host01. This class will evaluate 

986 the string and lookup the ip_address[0] from the host01 resource object 

987 and update the string with the correct information. This makes it helpful 

988 when orchestrate/execute tasks require data from the hosts itself. 

989 """ 

990 

991 def __init__(self, hosts): 

992 """Constructor. 

993 

994 :param hosts: teflo host resources 

995 :type hosts: list 

996 """ 

997 self.hosts = hosts 

998 

999 # regular expression to search for in the string 

1000 # data to be injected needs to be in the format of 

1001 # { host01.metadata.k1 } 

1002 self.regexp = r"\{(.*?)\}" 

1003 

1004 # regex to check jsonpath strings 

1005 self.exclusion_chk_str = r"^range|^[|.|$|@]|[\w|']+:" 

1006 

1007 def host_exist(self, node): 

1008 """Determine if the host defined in the string formatted var is valid. 

1009 

1010 In the case no host is found, an exception is raised. 

1011 

1012 :param node: node name 

1013 :type node: str 

1014 :return: teflo host resource matching based on node input 

1015 :rtype: object 

1016 """ 

1017 for host in self.hosts: 

1018 if node == getattr(host, 'name'): 

1019 return host 

1020 raise TefloError('Node %s not found!' % node) 

1021 

1022 def inject(self, command): 

1023 """Main worker. 

1024 

1025 This method will perform the data injection. 

1026 

1027 :param command: command to inject data into 

1028 :type command: str 

1029 :return: updated command 

1030 :rtype: str 

1031 """ 

1032 

1033 variables = list(map(str.strip, re.findall(self.regexp, command))) 

1034 

1035 if not variables.__len__(): 

1036 return command 

1037 

1038 for variable in variables: 

1039 if re.match(self.exclusion_chk_str, variable): 

1040 LOG.debug("JSONPath format was identified in the command %s." % variable) 

1041 continue 

1042 else: 

1043 value = None 

1044 _vars = variable.split('.') 

1045 node = _vars.pop(0) 

1046 

1047 # verify variable has a valid host set 

1048 host = self.host_exist(node) 

1049 

1050 for index, item in enumerate(_vars): 

1051 try: 

1052 # is the item intended to be a position in a list, if so 

1053 # get the key and position 

1054 key = item.split('[')[0] 

1055 pos = int(item.split('[')[1].split(']')[0]) 

1056 

1057 if value: 

1058 # get the latest value from the dictionary 

1059 value = value[key][pos] 

1060 else: 

1061 # get latest value from host 

1062 if hasattr(host, key) and index <= 0: 

1063 value = getattr(host, key)[pos] 

1064 if isinstance(value, str): 

1065 break 

1066 

1067 # is the value a dict, if so keep going! 

1068 if isinstance(value, dict): 

1069 continue 

1070 except IndexError: 

1071 # item is not intended to be a position in a list 

1072 

1073 # check if the item is an attribute of the host 

1074 if hasattr(host, item) and index <= 0: 

1075 value = getattr(host, item) 

1076 

1077 if isinstance(value, str): 

1078 # we know the value has no further traversing to do 

1079 break 

1080 # value is either a list or dict, more traversing to do 

1081 continue 

1082 else: 

1083 if value is None: 

1084 raise AttributeError('%s not found in host %s!' % 

1085 (item, getattr(host, 'name'))) 

1086 

1087 # check if the item's value is a dict and update the value 

1088 # for further traversing to do 

1089 try: 

1090 if isinstance(value[item], dict): 

1091 value = value[item] 

1092 continue 

1093 except KeyError: 

1094 raise TefloError('%s not found in %s' % (item, value)) 

1095 

1096 # final check to get value no more traversing required 

1097 if value: 

1098 value = value[item] 

1099 except KeyError: 

1100 raise TefloError('Unable to locate item %s!' % item) 

1101 

1102 command = command.replace('{ %s }' % variable, value) 

1103 return command

1173def find_artifacts_on_disk(data_folder, report_name, art_location=[]):

1176 to see if they exist. If the Execute collected artifacts, it will check the 

1177 the .results/artifacts/ specifically for the artifacts. 

1178

1305def validate_render_scenario(scenario, temp_data=None): 

1306 """ 

1307 This method takes the absolute path of the scenario descriptor file and returns back a list of 

1308 data streams of scenario(s) after doing the following checks: 

1309 (1) Checks there is no yaml.safe_load error for the provided scenario file 

1310 (2) Checks for include section present in the scenario file 

1311 (3) Checks the include section has valid scenario file path and it is not empty 

1312 (4) Checks there is no yaml.safe_load error for scenario file in the include section 

1313 :param scenario: scenario file path 

1314 :type scenario: str 

1315 :param temp_data: the file path to jinja template vars data or a json dictionary of vars data 

1316 :type temp_data: dict or str 

1317 :return: scenario data stream(s) 

1318 :rtype: list of data streams 

1319 """ 

1320 scenario_stream_list = list() 

1321 

1322 if temp_data: 

1323 if os.path.isfile(temp_data): 

1324 temp_data = file_mgmt('r', temp_data) 

1325 else: 

1326 temp_data = json.loads(temp_data) 

1327 # Updating th eos.environ to the variable data 

1328 temp_data.update(os.environ) 

1329 else: 

1330 # if temp_data isNone, sending only the os.environment 

1331 temp_data = os.environ 

1332 try: 

1333 data = yaml.safe_load(template_render(scenario, temp_data)) 

1334 # adding master scenario as the first scenario data stream 

1335 scenario_stream_list.append(template_render(scenario, temp_data)) 

1336 if 'include' in data.keys(): 

1337 include_item = data['include'] 

1338 include_template = list() 

1339 if include_item is not None: 

1340 for item in include_item: 

1341 if os.path.isfile(item): 

1342 item = os.path.abspath(item) 

1343 # check to verify the data in included scenario is valid 

1344 try: 

1345 yaml.safe_load(template_render(item, temp_data)) 

1346 include_template.append(template_render(item, temp_data)) 

1347 except yaml.YAMLError: 

1348 # raising Teflo error to differentiate the yaml issue is with included scenario 

1349 raise TefloError('Error loading updated included scenario data!') 

1350 else: 

1351 # raising HelperError if included file is invalid or included section is empty 

1352 raise HelpersError('Included File is invalid or Include section is empty .'

1361def ssh_key_file_generator(workspace, ssh_key_param): 

1362 """ 

1363 A method to find if the ssh key value in a provider 

1364 param is a path to a public or private key. If it 

1365 is a public key, just return the key back. If it 

1366 is a private key, generate the public key from it 

1367 and return that back. 

1368 

1369 Right now this is only really used by Linchpin 

1370 in the LinchpinResourceBuilder class for Beaker 

1371 resources because they expect a public key and the 

1372 various ssh key options in their resource defintions 

1373 

1374 :param workspace: the Teflo workspace keys directory 

1375 :type workspace: path as a string 

1376 :param ssh_key_param: the ssh_key from the provider param 

1377 :type ssh_key_param: value of the ssh_key param either a path or an actual key 

1378 :return: a path to a public key 

1379 """ 

1380 

1381 # setup absolute path for key 

1382 key = os.path.join(workspace, ssh_key_param) 

1383 

1384 # set permission of the key 

1385 try: 

1386 os.chmod(key, stat.S_IRUSR | stat.S_IWUSR) 

1387 except OSError as ex: 

1388 raise HelpersError( 

1389 'Error setting private key file permissions: %s' % ex 

1390 ) 

1391 

1392 # Lets assume it's a private key file and try to load it 

1393 # and create a public key for it 

1394 try: 

1395 rsa_key = RSAKey.from_private_key_file(key) 

1396 # generate public key from private 

1397 public_key = os.path.join( 

1398 workspace, ssh_key_param + ".pub" 

1399 ) 

1400 with open(public_key, 'w') as f: 

1401 f.write('%s %s\n' % (rsa_key.get_name(), rsa_key.get_base64())) 

1402 return public_key 

1403 except SSHException: 

1404 # Exception means the key file was invalid. 

1405 # Assume it's a public key and return it 

1406 return key

1508def create_individual_testrun_results(artifact_locations, config): 

1509 """this method creates a summary of total tests passed, failed, skipped for all the xml files found 

1510 as artifacts 

1511 :param artifact_locations: list of relative paths of artifacts where root dir is the key and artifact names are 

1512 values 

1513 :type artifact_locations: list 

1514 :param config: config parameter used by execute resource 

1515 :type config: dict 

1516 :return testruns: a dictionary of test results summary for individual xml files as well as aggregate of all xml 

1517 files found 

1518 :rtype testruns: dict 

1519 """ 

1520 fnd_paths = list() 

1521 individual_res = list() 

1522 # build the regex query to get only xml files 

1523 regquery = build_artifact_regex_query('*.xml') 

1524 # search the artifact location dictionary provided to search in the .results folder 

1525 fnd_paths.extend( 

1526 search_artifact_location_dict(artifact_locations, '*.xml', config.get('RESULTS_FOLDER'), 

1527 regquery)) 

1528 try: 

1529 for path in fnd_paths: 

1530 trun = dict() 

1531 trun['total_tests'] = 0 

1532 trun['failed_tests'] = 0 

1533 trun['skipped_tests'] = 0 

1534 trun['passed_tests'] = 0 

1535 tree = ET.parse(path) 

1536 root = tree.getroot() 

1537 temp = list() 

1538 # if root.tag is testsuites then collect all the element which are 'testsuite' 

1539 # if root.tag is testsuite then just collect that element 

1540 if root.tag == "testsuites": 

1541 temp.extend(root.findall('testsuite')) 

1542 elif root.tag == "testsuite": 

1543 temp.append(root) 

1544 if temp: 

1545 for test in temp: 

1546 trun['total_tests'] = trun['total_tests'] + len(test.findall('testcase')) 

1547 trun['failed_tests'] = trun['failed_tests'] + len([testcase.find('failure') 

1548 for testcase in 

1549 test.findall('testcase') 

1550 if 

1551 testcase.findall('failure')]) 

1552 trun['skipped_tests'] = trun['skipped_tests'] + len([testcase.find('skipped') 

1553 for testcase in 

1554 test.findall('testcase') 

1555 if testcase.findall( 

1556 'skipped')]) 

1557 trun['passed_tests'] = trun['total_tests'] - trun['failed_tests'] - trun[ 

1558 'skipped_tests'] 

1559 individual_res.append({os.path.basename(path): trun}) 

1560 else: 

1561 LOG.warning( 

1562 "The xml file %s does not have the correct format (no 'testsuite' or 'testsuites'" 

1563 " tags) to collect testrun results" % path) 

1564 continue 

1565 except ET.ParseError:
