95 def validate_opt_params(self, host): 

96 """Validate the optional parameters exists in the host resource. 

97 

98 :param host: host resource 

99 :type host: object 

100 """ 

101 if getattr(host, 'provisioner_plugin') is not None: 

102 provisioner_name = getattr(host, 'provisioner_plugin').__plugin_name__ 

103 else: 

104 provisioner_name = getattr(host, 'provisioner').__provisioner_name__ 

105 name = getattr(host, 'name') 

106 param_values = getattr(host, 'provider_params') 

107 self.validate_common_opt_params(name, provisioner_name, param_values) 

108 self.validate_opt_linchpin_params(name, provisioner_name, param_values) 

109 

110 def validate_common_opt_params(self, resource_name, provisioner_name, params): 

111 

112 for item in self.comm_opt_params: 

113 param, param_type = item[0], item[1] 

114 msg = "Resource %s : optional param '%s' " % (resource_name, param) 

115 try: 

116 param_value = params[param] 

117 self.logger.info(msg + 'exists.') 

118 

119 if not type(param_value) in param_type: 

120 self.logger.error( 

121 ' - Type=%s, Optional Type=%s. (ERROR)' % 

122 (type(param_value), param_type)) 

123 raise TefloProviderError( 

124 'Error occurred while validating required provider ' 

125 'parameters for resource %s' % resource_name 

126 ) 

127 except KeyError: 

128 self.logger.warning(msg + 'is undefined for resource.') 

129 

130 for item in self.linchpin_comm_opt_params: 

131 param, param_type = item[0], item[1] 

132 msg = "Resource %s : optional param '%s' " % (resource_name, param) 

133 err_msg = "Resource %s : optional param '%s' or it's value type is " \ 

134 "not support by the openstack-libcloud provisioner" % (resource_name, param) 

135 try: 

136 param_value = params[param] 

137 if 'linchpin' not in provisioner_name: 

138 self.logger.error(msg) 

139 raise TefloProviderError(err_msg) 

140 else: 

141 self.logger.info(msg + 'exists.') 

142 

143 if not type(param_value) in param_type: 

144 self.logger.error( 

145 ' - Type=%s, Optional Type=%s. (ERROR)' % 

146 (type(param_value), param_type)) 

147 raise TefloProviderError( 

148 'Error occurred while validating required provider ' 

149 'parameters for resource %s' % resource_name 

150 ) 

151 except KeyError: 

152 self.logger.warning(msg + 'is undefined for resource.') 

153 

154 for item in self.teflo_comm_opt_params: 

155 param, param_type = item[0], item[1] 

156 msg = "Resource %s : optional param '%s' " % (resource_name, param) 

157 try: 

158 param_value = params[param] 

159 if 'linchpin' in provisioner_name: 

160 self.logger.warning( 

161 msg + 'is forward compatible with the linchpin provisioner. It will be translated ' 

162 'to be in proper format for linchpin.') 

163 if not type(param_value) in param_type: 

164 self.logger.warning( 

165 ' - Type=%s, Optional Type=%s. (WARNING)' % 

166 (type(param_value), param_type)) 

167 else: 

168 self.logger.info(msg + 'exists.') 

169 

170 if 'linchpin' in provisioner_name: 

171 if not type(param_value) in param_type: 

172 self.logger.warning( 

173 ' - Type=%s, Optional Type=%s. (WARNING)' % 

174 (type(param_value), param_type)) 

175 else: 

176 

177 if not type(param_value) in param_type: 

178 self.logger.error( 

179 ' - Type=%s, Optional Type=%s. (ERROR)' % 

180 (type(param_value), param_type)) 

181 raise TefloProviderError( 

182 'Error occurred while validating required provider ' 

183 'parameters for resource %s' % resource_name 

184 ) 

185 

186 except KeyError: 

187 self.logger.warning(msg + 'is undefined for resource.')
