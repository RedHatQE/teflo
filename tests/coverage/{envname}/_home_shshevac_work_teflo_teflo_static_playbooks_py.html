189ADHOC_SHELL_PLAYBOOK = ''' 

190- name: run shell and fetch results 

191 hosts: "{{ hosts }}" 

192 

193 tasks: 

194 - name: shell command 

195 shell: "{{ xcmd }}" 

196 register: sh_results 

197 ignore_errors: true 

198 {{ args }} 

199 {{ options }} 

200 

201 - name: to get correct (stderr/stdout/msg) output msg 

202 set_fact: 

203 sh_out: "{{ sh_results.stderr | default( sh_results.stdout )| default( sh_results.msg )| 

204 default('stderr,stdout.msg NOT present in the output') }}" 

205 - name: setting json str 

206 set_fact: 

207 json_str: "{ 'host_name': '{{ ansible_facts.hostname}}', 'rc': '{{ sh_results.rc | default(1) }}', 

208 'err': '{{ sh_out | regex_replace('\\\\s\\\\s+','')| regex_replace('\\n','') 

209 | regex_replace(\\"'\\",'') }}' }" 

210 - name: copy to shell results to a json file 

211 copy: 

212 content: "{{ ansible_play_hosts | map('extract', hostvars, 'json_str') | list | to_nice_json }}" 

213 dest: ./{{ 'shell-results-' + uuid }}.json 

214 run_once: true 

215 delegate_to: localhost 

216'''
