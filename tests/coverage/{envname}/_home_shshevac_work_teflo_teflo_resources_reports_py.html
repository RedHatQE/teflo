46 def __init__(self, 

47 config=None, 

48 name=None, 

49 importer=None, 

50 parameters={}, 

51 validate_task_cls=ValidateTask, 

52 report_task_cls=ReportTask, 

53 **kwargs): 

54 """Constructor. 

55 

56 :param config: teflo configuration 

57 :type config: dict 

58 :param name: report resource name 

59 :type name: str 

60 :param parameters: content which makes up the report resource 

61 :type parameters: dict 

62 :param validate_task_cls: teflos validate task class 

63 :type validate_task_cls: object 

64 :param report_task_cls: teflos report task class 

65 :type report_task_cls: object 

66 :param kwargs: additional key:value(s) 

67 :type kwargs: dict 

68 """ 

69 super(Report, self).__init__(config=config, name=name, **kwargs) 

70 

71 # set the timeout for VALIDATE 

72 try: 

73 if parameters.get('validate_timeout') is not None: 

74 self._validate_timeout = parameters.pop("validate_timeout") 

75 else: 

76 self._validate_timeout = config["TIMEOUT"]["VALIDATE"] 

77 except TypeError: 

78 self.logger.error("No teflo.cfg found, so no timeout will be set") 

79 self._validate_timeout = 0 

80 

81 # set the timeout for report 

82 try: 

83 if parameters.get('report_timeout') is not None: 

84 self._report_timeout = parameters.pop("report_timeout") 

85 else: 

86 self._report_timeout = config["TIMEOUT"]["REPORT"] 

87 except TypeError: 

88 self.logger.error("No teflo.cfg found, so no timeout will be set") 

89 self._report_timeout = 0 

90 

91 # set the report resource name 

92 if name is None: 

93 self._name = parameters.pop('name', None) 

94 if self._name is None: 

95 raise TefloReportError('Unable to build report object. Name' 

96 ' field missing!') 

97 else: 

98 self._name = name 

99 

100 # set the report description 

101 self._description = parameters.pop('description', None) 

102 

103 # set the execute that collected the artifacts 

104 self._executes = parameters.pop('executes', []) 

105 

106 # convert the executes into list format if executes defined as str format 

107 if self._executes is not None and isinstance(self._executes, string_types): 

108 self._executes = [val.strip() for val in self._executes.split(',')] 

109 

110 importer_name = parameters.pop('importer', importer) 

111 

112 parameters = self.__set_provider_attr__(parameters) 

113 

114 # Finally check and load external plugin implementation 

115 # if provider key was provided and importer_name was not provided , provider_name is used as importer name and 

116 # external importer plugins are searched used this name 

117 # if provider is not used importer name is must 

118 if importer_name is None: 

119 if self.has_provider: 

120 importer_name = self.provider 

121 else: 

122 raise TefloReportError('Importer or Provider name has to be provided') 

123 

124 found_name = False 

125 for name in get_importers_plugin_list(): 

126 if name.startswith(importer_name): 

127 found_name = True 

128 if found_name: 

129 self._importer_plugin = get_importer_plugin_class(importer_name) 

130 self.do_import = True 

131 else: 

132 self.logger.error('Importer %s for report artifacts %s is invalid.' 

133 % (importer_name, self.name)) 

134 sys.exit(1) 

135 

136 self._import_results = parameters.pop('import_results', []) 

137 

138 # set labels 

139 setattr(self, 'labels', parameters.pop('labels', [])) 

140 

141 # set the teflo task classes for the resource 

142 self._validate_task_cls = validate_task_cls 

143 self._report_task_cls = report_task_cls 

144 

145 # reload construct task methods 

146 self.reload_tasks() 

147 

148 # load the parameters into the object itself 

149 if parameters: 

150 self.load(parameters)

152 def __set_provider_attr__(self, parameters): 

153 """Configure the report provider attributes. 

154 

155 :param parameters: content which makes up the report resource 

156 :type parameters: dict 

157 :return: updated parameters 

158 :rtype: dict""" 

159 

160 # this is for backward compatibility, in case the provider key is used in the SDF 

161 self._provider = {} 

162 if parameters.get('provider', False): 

163 creds = parameters.get('provider').pop('credential', None) 

164 for p, v in parameters.pop('provider').items(): 

165 if p == 'name': 

166 self._provider = v 

167 continue 

168 setattr(self, p, v) 

169 self.has_provider = True 

170 else: 

171 # set no provider object 

172 creds = parameters.pop('credential', None) 

173 for p, v in parameters.items(): 

174 setattr(self, p, v) 

175 del self.provider 

176 self.has_provider = False 

177 

178 # lets set the credentials if any 

179 if creds: 

180 for item in self.config['CREDENTIALS']: 

181 if item['name'] == creds: 

182 self._credential = item 

183 break 

184 return parameters

295 def profile(self): 

296 """Builds a profile for the report resource. 

297 

298 :return: the report profile 

299 :rtype: OrderedDict 

300 """ 

301 filtered_attr = {k: v for k, v in vars(self).items() if not k.startswith('_') and k not in 

302 ['do_import', 'has_provider', 'all_hosts']} 

303 profile = OrderedDict() 

304 profile['name'] = self.name 

305 profile['description'] = self.description 

306 if not isinstance(self.importer_plugin, string_types): 

307 profile['importer'] = getattr( 

308 self.importer_plugin, '__plugin_name__') 

309 else: 

310 profile['importer'] = self.importer_plugin 

311 if hasattr(self, 'provider') and len(getattr(self, 'provider', {})) != 0: 

312 profile.update(OrderedDict(provider={})) 

313 profile.get('provider').update(name=self.provider) 

314 profile.get('provider').update(credential=self.credential.get('name')) 

315 profile.get('provider').update(filtered_attr) 

316 else: 

317 profile.update(credential=self.credential.get('name')) 

318 profile.update(filtered_attr) 

319 

320 # set the labels for report resource 

321 profile['labels'] = self.labels 

322 

323 # set the report's executes 

324 if all(isinstance(item, string_types) for item in self.executes): 

325 profile.update(executes=[execute for execute in self.executes]) 

326 else: 

327 profile.update(dict(executes=[execute.name for execute in self.executes])) 

328 

329 # set the report's import results 

330 profile.update({'import_results': self.import_results}) 

331 

332 return profile
