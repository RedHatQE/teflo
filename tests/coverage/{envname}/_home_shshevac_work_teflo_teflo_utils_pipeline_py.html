49class PipelineBuilder(object): 

50 """ 

51 The primary class for building teflos pipelines to execute. A pipeline 

52 will exists for each teflo task. Within each pipeline consists of all 

53 resources for that specific task to process. 

54 """ 

55 

56 def __init__(self, name): 

57 """Constructor. 

58 

59 :param name: teflo task name 

60 :type name: str 

61 """ 

62 self._name = name 

63 

64 # pipelines are a tuple data structure consisting of a name, type and 

65 # list of tasks 

66 self.pipeline_template = namedtuple( 

67 'Pipeline', ('name', 'type', 'tasks')) 

68 

69 @property 

70 def name(self): 

71 """Return the pipeline name""" 

72 return self._name 

73 

74 def is_task_valid(self): 

75 """Check if the pipeline task name is valid for teflo. 

76 

77 :return: whether task is valid or not. 

78 :rtype: bool 

79 """ 

80 try: 

81 TASKLIST.index(self.name) 

82 except ValueError: 

83 return False 

84 return True 

85 

86 def task_cls_lookup(self): 

87 """Lookup the pipeline task class type. 

88 

89 :return: the class associated for the pipeline task. 

90 :rtype: class 

91 """ 

92 for cls in get_core_tasks_classes(): 

93 if cls.__task_name__ == self.name: 

94 return cls 

95 raise TefloError('Unable to lookup task %s class.' % self.name) 

96 

97 def build(self, scenario, teflo_options): 

98 """Build teflo pipeline. 

99 

100 This method first collects scenario tasks and resources for each scenario(child and master). It filters out 

101 specific resources for the scenario based on if labels or skip_labels are provided in teflo options. 

102 If no labels/ skip_labels are provided all resources for that scenario are picked. 

103 Then for each of the resource/scenario task the method checks if that resource/scenario tasks has any tasks 

104 with name matching the name for self.task(the task for which the pipeline is getting built). If it has then that 

105 tasks gets added to the pipeline and that gets returned 

106 

107 :param scenario: teflo scenario object containing all scenario 

108 data. 

109 :type scenario: scenario object 

110 :param teflo_options: extra options provided during teflo run 

111 :type teflo_options: dict 

112 :return: teflo pipeline to run for the given task for all the scenarios 

113 :rtype: namedtuple 

114 """ 

115 

116 # pipeline init 

117 pipeline = self.pipeline_template( 

118 self.name, 

119 self.task_cls_lookup(), 

120 list() 

121 ) 

122 

123 scenario_get_tasks = list() 

124 # Get all tasks for the scenario and its included scenarios 

125 if scenario.child_scenarios: 

126 for sc in scenario.child_scenarios: 

127 scenario_get_tasks.extend([item for item in getattr(sc, 'get_tasks')()]) 

128 # only master scenario no child scenarios 

129 scenario_get_tasks.extend([item for item in getattr(scenario, 'get_tasks')()]) 

130 

131 # Collecting resources based on task type 

132 if self.name.lower() in ['validate', 'provision', 'cleanup']: 

133 # scenario resource 

134 for task in scenario_get_tasks: 

135 if task['task'].__task_name__ == self.name: 

136 pipeline.tasks.append(set_task_class_concurrency(task, task['resource'])) 

137 

138 # asset resource filtered based on labels 

139 for asset in filter_resources_labels(scenario.get_all_assets(), teflo_options): 

140 for task in asset.get_tasks(): 

141 if task['task'].__task_name__ == self.name: 

142 pipeline.tasks.append(set_task_class_concurrency(task, asset)) 

143 

144 if self.name.lower() in ['validate', 'orchestrate', 'cleanup']: 

145 # action resource 

146 # get action resource based on if its status 

147 # check if cleanup task do NOT filter by status 

148 if self.name != 'cleanup': 

149 scenario_actions = filter_actions_on_failed_status(scenario.get_all_actions()) 

150 else: 

151 scenario_actions = scenario.get_all_actions() 

152 # action resource filtered based on labels 

153 for action in filter_resources_labels(scenario_actions, teflo_options): 

154 for task in action.get_tasks(): 

155 if task['task'].__task_name__ == self.name: 

156 # fetch & set hosts for the given action task 

157 task = fetch_assets(scenario.get_all_assets(), task) 

158 pipeline.tasks.append(set_task_class_concurrency(task, action)) 

159 

160 if self.name.lower() in ['validate', 'execute']: 

161 # execute resource filtered based on labels 

162 for execute in filter_resources_labels(scenario.get_all_executes(), teflo_options): 

163 for task in execute.get_tasks(): 

164 if task['task'].__task_name__ == self.name: 

165 # fetch & set hosts for the given executes task 

166 task = fetch_assets(scenario.get_all_assets(), task) 

167 pipeline.tasks.append(set_task_class_concurrency(task, execute)) 

168 

169 if self.name.lower() in ['validate', 'report']: 

170 # report resource filtered based on labels 

171 for report in filter_resources_labels(scenario.get_all_reports(), teflo_options): 

172 for task in report.get_tasks(): 

173 if task['task'].__task_name__ == self.name: 

174 # fetch & set hosts and executes for the given reports task 

175 task = fetch_executes(scenario.get_all_executes(), scenario.get_all_assets(), task) 

176 pipeline.tasks.append(set_task_class_concurrency(task, report)) 

177 

178 if self.name.lower() in ['validate']: 

179 # notification resource 

180 for notification in filter_notifications_to_skip(scenario.get_all_notifications(), teflo_options): 

181 for task in notification.get_tasks(): 

182 if task['task'].__task_name__ == self.name: 

183 task['resource'].scenario = scenario 

184 pipeline.tasks.append(task) 

185 

186 # reverse the order of the tasks to be executed for cleanup task 

187 if self.name == CleanupTask.__task_name__: 

188 pipeline.tasks.reverse() 

189 

190 return pipeline

214 return True

216 def build(self, scenario, teflo_options): 

217 """Build teflo notification pipeline. 

218 

219 This method first collects scenario tasks and resources for each scenario(child and master). It filters out 

220 specific resources for the scenario based on if labels or skip_labels are provided in teflo options. 

221 If no labels/ skip_labels are provided all resources for that scenario are picked. 

222 Then for each of the resource/scenario task the method checks if that resource/scenario tasks has any tasks 

223 with name matching the name for self.task(the task for which the pipeline is getting built). If it has then that 

224 tasks gets added to the pipeline and that gets returned 

225 

226 :param scenario: teflo scenario object containing all scenario 

227 data. 

228 :type scenario: scenario object 

229 :param teflo_options: extra options provided during teflo run 

230 :type teflo_options: dict 

231 :return: teflo notification pipeline to run for the given task for all the scenarios 

232 :rtype: namedtuple 

233 """ 

234 

235 # pipeline init 

236 pipeline = self.pipeline_template( 

237 self.name, 

238 self.task_cls_lookup(), 

239 list() 

240 ) 

241 

242 # get notifications 

243 scenario_notifications = [item for item in filter_notifications_to_skip(scenario.get_all_notifications(), 

244 teflo_options)] 

245 scenario_notifications = [item for item in 

246 filter_notifications_on_trigger(self.trigger, scenario_notifications, 

247 getattr(scenario, 'passed_tasks'), 

248 getattr(scenario, 'failed_tasks')) 

249 ] 

250 

251 # notification resource 

252 for notification in scenario_notifications: 

253 for task in notification.get_tasks(): 

254 if task['task'].__task_name__ == self.name: 

255 task['resource'].scenario = scenario 

256 pipeline.tasks.append(task) 

257 

258 return pipeline
