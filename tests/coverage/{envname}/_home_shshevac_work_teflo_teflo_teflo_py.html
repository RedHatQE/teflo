42class Teflo(LoggerMixin, TimeMixin): 

43 """ 

44 The Teflo object acts as the central object. We call this object 

45 'the teflo compound'. Like in chemistry, a teflo molecule helps on the 

46 construction of compounds of other molecules. We can use this analogy 

47 to understand how Teflo object creates compounds of 'resources'. 

48 

49 The valid resources for Teflo can be found at ~teflo.resources package. 

50 These resources have special connection with teflo, which allows teflo 

51 to run the resources tasks, without the need to understand each resource 

52 in depth. Think of all caron resources as instances that are blessed by 

53 Teflo object. A teflo compound can have as many resource as it can. 

54 

55 Once a compound of resources is defined, the function ~self.run will 

56 take care of running all the tasks needed to have the compound of 

57 resources up and running. 

58 

59 The Teflo object passes through 6 main stages: 

60 (see all types of tasks at ~teflo.tasks package) 

61 

62 1. Validation - a pipeline of tasks instances of ValidationTask. 

63 2. Provision - a pipeline of tasks instances of ProvisionTask. 

64 3. Orchestrate - a pipeline of tasks instances of OrchestrateTask. 

65 4. Execute - a pipeline of tasks instances of ExecuteTask. 

66 5. Report - a pipeline of tasks instances of ReportTask. 

67 6. Cleanup - a pipeline of tasks instances of CleanupTask. 

68 

69 Each resource can have its own set of tasks. Theses tasks will be 

70 loaded within the central pipeline, the ~self.pipelines object. 

71 """

247 def load_from_yaml(self, filedata): 

248 """ 

249 Given the YAML filename with the scenario description, this 

250 function will load all resources described in the file with 

251 the teflo object. 

252 

253 The main sections of the YAML descriptor (provision, orchestrate, 

254 execute and report) will be taken of the data in its respective 

255 data object. All the rest of the attributes within the YAML file 

256 will be loaded into the ~self.scenario object. 

257 

258 If include section for included scenarios is present then they get 

259 added as child scenario objects for the main scenario(~self.scenario). 

260 These scenarios in turn go through the same process of taking out the 

261 main sections from the data in its respective data object 

262 

263 The additional attributes within the file will be ignored by the Scenario 

264 resource. 

265 

266 Once each data section is taken of the main descriptor, it will 

267 then be be loaded in each respective lists within the 

268 ~self.scenario object. 

269 

270 :param filedata: list of full data object for the YAML file descriptor 

271 :return: 

272 """ 

273 # setting up master scenario 

274 self.scenario.yaml_data = filedata[0] 

275 self._populate_scenario_resources(self.scenario, filedata[0]) 

276 

277 # creating child scenario objects 

278 if len(filedata) > 1: 

279 for scenario_stream in filedata[1:]: 

280 ch_scenario = Scenario(config=self.config) 

281 ch_scenario.yaml_data = scenario_stream 

282 self._populate_scenario_resources(ch_scenario, scenario_stream) 

283 self.scenario.add_child_scenario(ch_scenario)

350 def run(self, tasklist=TASKLIST): 

351 """ 

352 This function assumes there are zero or more tasks to be 

353 loaded within the list of pipelines: ~self.pipelines. 

354 

355 It will run through all resources within the ~self.scenario 

356 object (lists of hosts, actions, executes, reports), including 

357 tasks associated with the ~self.scenario itself, and find if any 

358 of them has a task to be loaded in the pipelines. 

359 

360 Once a task is found, it is loaded within its respective 

361 pipeline and then each pipeline is sent to blaster blastoff. 

362 For every pipeline within ~self.pipelines, 

363 """ 

364 # lists to control which tasks passed or failed 

365 passed_tasks = list() 

366 failed_tasks = list() 

367 

368 # initialize overall status 

369 status = 0 

370 

371 # save start time 

372 self.start() 

373 

374 self._print_header(tasklist) 

375 

376 try: 

377 for task in sort_tasklist(tasklist): 

378 self.logger.info(' * Task : %s' % task) 

379 

380 # initially update list of passed tasks 

381 passed_tasks.append(task) 

382 if not self.teflo_options.get('no_notify', False): 

383 self.notify('on_start', status, passed_tasks, failed_tasks) 

384 data = self._run_pipeline(task) 

385 

386 # reload resource objects 

387 self.scenario.reload_resources(data) 

388 

389 if self.scenario.child_scenarios: 

390 [sc.reload_resources(data) for sc in self.scenario.child_scenarios] 

391 

392 # Creating inventory only when task is provision 

393 if task == 'provision': 

394 try: 

395 # create the master inventory 

396 for host in self.scenario.get_all_assets(): 

397 if (hasattr(host, 'role') or hasattr(host, 'groups')) and hasattr(host, 'ip_address'): 

398 self.logger.info('Populating master inventory file with host(s) %s' 

399 % getattr(host, 'name')) 

400 

401 self.cbn_inventory.create_master(all_hosts=self.scenario.get_all_assets()) 

402 except Exception as ex: 

403 raise TefloError("Error while creating the master inventory %s" % ex) 

404 

405 self.logger.info("." * 50) 

406 except Exception as ex: 

407 # set overall status 

408 status = 1 

409 

410 # pop task from passed list since it failed 

411 passed_tasks.pop(-1) 

412 

413 # update list of failed tasks 

414 failed_tasks.append(task) 

415 

416 self.logger.error(ex) 

417 

418 # reload resource objects 

419 self.scenario.reload_resources(ex.results) 

420 

421 if self.scenario.child_scenarios: 

422 [sc.reload_resources(ex.results) for sc in self.scenario.child_scenarios] 

423 

424 # roll back by cleaning up any resources that might have been provisioned 

425 if 'cleanup' in tasklist and [item for item in failed_tasks if item != 'cleanup']: 

426 if [item for item in passed_tasks if item == 'provision'] \ 

427 or [item for item in failed_tasks if item == 'provision']: 

428 self.logger.info("\n\n") 

429 self.logger.warning("A failure occurred before running the cleanup task. " 

430 "Attempting to run the cleanup task to roll back all provisioned resources.") 

431 task = tasklist[tasklist.index('cleanup')] 

432 

433 try: 

434 

435 data = self._run_pipeline(task) 

436 

437 # reload resource objects 

438 self.scenario.reload_resources(data) 

439 if self.scenario.child_scenarios: 

440 [sc.reload_resources(data) for sc in self.scenario.child_scenarios] 

441 passed_tasks.append(task) 

442 except Exception as ex: 

443 self.logger.error(ex) 

444 self.logger.error("There was a problem running the cleanup task to roll back the " 

445 "resources. You may need to manually cleanup any provisioned resources") 

446 failed_tasks.append(task) 

447 raise 

448 finally: 

449 # save end time 

450 self.end() 

451 # determine state 

452 state = 'FAILED' if status else 'PASSED' 

453 

454 # finally send out any notifications 

455 if not self.teflo_options.get('no_notify', False): 

456 self.notify('on_complete', status, passed_tasks, failed_tasks) 

457 

458 self._write_out_results() 

459 

460 self._print_footer(passed_tasks, failed_tasks, state) 

461 

462 self._archive_results() 

463 

464 sys.exit(status)

466 def notify(self, task, status=0, passed_tasks=None, failed_tasks=None): 

467 

468 setattr(self.scenario, 'overall_status', status) 

469 setattr(self.scenario, 'passed_tasks', []) 

470 setattr(self.scenario, 'failed_tasks', []) 

471 

472 if passed_tasks: 

473 setattr(self.scenario, 'passed_tasks', passed_tasks) 

474 

475 if failed_tasks: 

476 setattr(self.scenario, 'failed_tasks', failed_tasks) 

477 

478 if task == 'on_demand': 

479 self.start() 

480 self._print_header(['notify']) 

481 self.logger.info(' * Task : notify') 

482 

483 # blast off the pipeline list of tasks 

484 try: 

485 

486 self.logger.info('Sending out any notifications that are registered.') 

487 data = self._run_pipeline(task) 

488 

489 # reload resource objects 

490 self.scenario.reload_resources(data) 

491 

492 if self.scenario.child_scenarios: 

493 [sc.reload_resources(data) for sc in self.scenario.child_scenarios] 

494 except Exception as ex: 

495 status = 1 

496 self.logger.error(ex) 

497 self.logger.error('One or more notifications failed. Refer to the scenario.log') 

498 

499 # reload resource objects 

500 self.scenario.reload_resources(ex.results) 

501 

502 if self.scenario.child_scenarios: 

503 [sc.reload_resources(ex.results) for sc in self.scenario.child_scenarios] 

504 finally: 

505 if task == 'on_demand': 

506 # save end time 

507 self.end() 

508 # determine state 

509 state = 'FAILED' if status else 'PASSED' 

510 

511 self._write_out_results() 

512 

513 self._print_footer(getattr(self.scenario, 'passed_tasks'), 

514 getattr(self.scenario, 'failed_tasks'), 

515 state) 

516 

517 self._archive_results() 

518 

519 sys.exit(status)

521 def _run_pipeline(self, task): 

522 

523 data = {} 

524 

525 # create a pipeline builder object 

526 pipe_builder = PipelineFactory.get_pipeline(task) 

527 

528 # check if teflo supports the task 

529 if not pipe_builder.is_task_valid(): 

530 self.logger.warning('Task %s is not valid by teflo.', task) 

531 return data 

532 

533 pipeline = pipe_builder.build(self.scenario, self.teflo_options) 

534 

535 self.logger.info('.' * 50) 

536 self.logger.info('Starting tasks on pipeline: %s', 

537 pipeline.name) 

538 

539 # check if pipeline has tasks to be run 

540 if not pipeline.tasks: 

541 self.logger.warning('... no tasks to be executed ...') 

542 return data 

543 

544 # create blaster object with pipeline to run 

545 blast = blaster.Blaster(pipeline.tasks) 

546 

547 # blast off the pipeline list of tasks reload_resources 

548 data = blast.blastoff( 

549 serial=not pipeline.type.__concurrent__, 

550 raise_on_failure=True 

551 ) 

552 

553 return data
