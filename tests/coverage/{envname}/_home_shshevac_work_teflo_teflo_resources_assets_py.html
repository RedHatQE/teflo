40class Asset(TefloResource): 

41 """ 

42 The asset resource class. The teflo compound can contain x amount of hosts or 

43 other asset types (i.e. virtual networks, storage, etc.). 

44 Their primary responsibility is to define details about the system resource 

45 to be created in their declared provider. Along with saving information 

46 such as (ip addresses, etc) for use by the action or execute compounds of 

47 teflo. 

48 """

79

83 :type name: str 

84 :param provisioner: provisioner name used to create the host in the

94 :type cleanup_task_cls: object 

95 :param kwargs: additional key:value(s) 

96 :type kwargs: dict 

97 """ 

98 super(Asset, self).__init__(config=config, name=name, **kwargs)

100 # set the timeout for VALIDATE 

101 try: 

102 if parameters.get('validate_timeout') is not None: 

103 self._validate_timeout = parameters.pop("validate_timeout") 

104 else: 

105 self._validate_timeout = config["TIMEOUT"]["VALIDATE"] 

106 except TypeError: 

107 self.logger.error("No teflo.cfg found, so no timeout will be set") 

108 self._validate_timeout = 0 

109 

110 # set the timeout for provsion 

111 try: 

112 if parameters.get('provision_timeout') is not None: 

113 self._provision_timeout = parameters.pop("provision_timeout") 

114 else: 

115 self._provision_timeout = config["TIMEOUT"]["PROVISION"] 

116 except TypeError: 

117 self.logger.error("No teflo.cfg found, so no timeout will be set") 

118 self._provision_timeout = 0 

119 # set the timeout for cleanup 

120 try: 

121 if parameters.get('cleanup_timeout') is not None: 

122 self._cleanup_timeout = parameters.pop("cleanup_timeout") 

123 else: 

124 self._cleanup_timeout = config["TIMEOUT"]["CLEANUP"] 

125 except TypeError: 

126 self.logger.error("No teflo.cfg found, so no timeout will be set") 

127 self._cleanup_timeout = 0 

128 

129 # set name attribute & apply filter 

130 if name is None: 

131 self._name = parameters.pop('name', None) 

132 if self._name is None: 

133 self._name = 'hst{0}'.format(gen_random_str(10)) 

134 else: 

135 self._name = name 

136 

137 # set description attribute 

138 self._description = parameters.pop('description', None) 

139 

140 # preset role/groups attribute 

141 self._role = None 

142 self._groups = None 

143 try: 

144 # convert the groups into list format if groups defined as str format 

145 self._groups = parameters.pop('groups') 

146 if isinstance(self._groups, string_types): 

147 self._groups = self._groups.replace(' ', '').split(',') 

148 del self.role 

149 except KeyError: 

150 try: 

151 # convert the roles into list format if roles defined as str format 

152 self._role = parameters.pop('role') 

153 if isinstance(self._role, string_types): 

154 self._role = self._role.replace(' ', '').split(',') 

155 self.logger.warning('A role has been found. This parameter will be deprecated' 

156 ' it is recommend to change over to groups.') 

157 del self.groups 

158 except KeyError: 

159 self.logger.warning('A group or a role was not set for asset %s.' % self.name) 

160 del self.role, self.groups 

161 

162 # set metadata attribute (data pass-through) 

163 self._metadata = parameters.pop('metadata', {}) 

164 

165 # set ansible parameters 

166 self._ansible_params = parameters.pop('ansible_params', {}) 

167 

168 # check if ansible_params and roles groups is empty 

169 if not hasattr(self, 'role') or not hasattr(self, 'groups') \ 

170 and not self._ansible_params: 

171 self.no_inventory = True 

172 

173 # Pop any attributes that aren't needed like data_folder and workspace 

174 # since that already gets set. 

175 parameters.pop('workspace', None) 

176 parameters.pop('data_folder', None) 

177 

178 setattr(self, 'labels', parameters.pop('labels', [])) 

179 

180 # determine if the host is a static machine already provisioned 

181 # how? if the ip_address param is defined and provider & provisioner is not defined 

182 # then we can be safe to say the machine is static 

183 if 'ip_address' in parameters and 'provider' not in parameters and \ 

184 'provisioner' not in parameters: 

185 self._ip_address = parameters.pop('ip_address') 

186 # set flag to control whether the host is static or not 

187 self.is_static = True 

188 self._provisioner = None 

189 del self.provisioner 

190 else: 

191 # set flag to control whether the host is static or not 

192 self.is_static = False 

193 

194 # lets get the right provisioner to use 

195 provisioner_name = parameters.pop('provisioner', provisioner) 

196 

197 # save ip address if one is provided 

198 self._ip_address = parameters.pop('ip_address', None) 

199 

200 # delete the ip_addres prop if none provided 

201 if not self.ip_address: 

202 del self.ip_address 

203 

204 # host needs to be provisioned, get the provider parameters 

205 parameters = self.__set_provider_attr_(parameters) 

206 

207 # lets setup any feature toggles that we defined in the configuration file 

208 self.__set_feature_toggles_() 

209 

210 # TODO remove this , it is additional 

211 self._provisioner = get_default_provisioner_plugin() 

212 

213 if provisioner_name is None and self.has_provider: 

214 try: 

215 self._provisioner = get_default_provisioner_plugin(self._provider) 

216 except KeyError: 

217 raise TefloResourceError('Specified provider is not supported by a provisioner.') 

218 elif provisioner_name: 

219 found_name = False 

220 for name in get_provisioners_plugins_list(): 

221 if name.startswith(provisioner_name): 

222 found_name = True 

223 break 

224 

225 if found_name: 

226 if self.has_provider and \ 

227 is_provider_mapped_to_provisioner(self._provider, provisioner_name): 

228 self._provisioner = get_provisioner_plugin_class(provisioner_name) 

229 else: 

230 self._provisioner = get_provisioner_plugin_class(provisioner_name) 

231 else: 

232 self.logger.error('Provisioner %s for asset %s is invalid.' 

233 % (provisioner_name, self.name)) 

234 raise TefloResourceError('The specified provisioner is not valid for the asset type.') 

235 else: 

236 raise TefloResourceError('Could not find a provisioner to satisfy the asset type.') 

237 

238 # set the teflo task classes for the resource 

239 self._validate_task_cls = validate_task_cls 

240 self._provision_task_cls = provision_task_cls 

241 self._cleanup_task_cls = cleanup_task_cls

257 def __set_provider_attr_(self, parameters): 

258 """Configure the host provider attributes. 

259 

260 :param parameters: content which makes up the host resource 

261 :type parameters: dict 

262 :return: updated parameters 

263 :rtype: dict 

264 """ 

265 

266 self._provider = {} 

267 

268 if parameters.get('provider', False): 

269 creds = parameters.get('provider').pop('credential', None) 

270 for p, v in parameters.pop('provider').items(): 

271 if p == 'name': 

272 self._provider = v 

273 continue 

274 setattr(self, p, v) 

275 self.has_provider = True 

276 else: 

277 # set no provider object 

278 creds = parameters.pop('credential', None) 

279 for p, v in parameters.items(): 

280 setattr(self, p, v) 

281 del self.provider 

282 self.has_provider = False 

283 

284 # lets set the credentials if any 

285 if creds: 

286 for item in self.config['CREDENTIALS']: 

287 if item['name'] == creds: 

288 self._credential = item 

289 break 

290 

291 return parameters

518 def profile(self): 

519 """Builds a profile for the host resource. 

520 

521 :return: the host profile 

522 :rtype: OrderedDict 

523 """ 

524 profile = OrderedDict() 

525 filtered_attr = {k: v for k, v in vars(self).items() if not k.startswith('_') and k not in 

526 ['is_static', 'has_provider', 'no_inventory']} 

527 

528 # update asset fields 

529 for f in self._fields: 

530 if f == 'role' and hasattr(self, f): 

531 if all(isinstance(item, string_types) for item in self.role): 

532 profile.update(role=[role for role in self.role]) 

533 continue 

534 elif f == 'groups' and hasattr(self, f): 

535 if all(isinstance(item, string_types) for item in self.groups): 

536 profile.update(groups=[group for group in self.groups]) 

537 continue 

538 elif f == 'provisioner' and hasattr(self, f): 

539 profile.update({'provisioner': getattr( 

540 self.provisioner, '__plugin_name__')}) 

541 if hasattr(self, 'provider') and len(getattr(self, 'provider', {})) != 0: 

542 profile.update(OrderedDict(provider={})) 

543 profile.get('provider').update(name=self.provider) 

544 profile.get('provider').update(filtered_attr) 

545 else: 

546 profile.update(filtered_attr) 

547 continue 

548 elif f == 'credential' and hasattr(self, f): 

549 if len(getattr(self, 'provider', {})) == 0: 

550 profile.update(credential=self.credential.get('name')) 

551 else: 

552 profile.get('provider').update(credential=self.credential.get('name')) 

553 continue 

554 elif f == 'metadata' and hasattr(self, f): 

555 if len(getattr(self, f)) != 0: 

556 profile.update({f: getattr(self, f)}) 

557 continue 

558 elif hasattr(self, f) and getattr(self, f) is not None: 

559 profile.update({f: getattr(self, f)}) 

560 

561 return profile
