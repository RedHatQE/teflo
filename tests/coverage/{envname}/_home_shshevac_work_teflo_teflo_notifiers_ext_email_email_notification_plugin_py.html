62 def send_message(self): 

63 """ 

64 Send the message. 

65 """ 

66 smtp = '' 

67 try: 

68 smtp_host = self.creds_params.get('smtp_host') 

69 if self.creds_params.get('smtp_port', False): 

70 smtp_host = ':'.join([smtp_host, self.creds_params.get('smtp_port')]) 

71 

72 self.logger.debug("Connecting to %s", smtp_host) 

73 

74 smtp = smtplib.SMTP(smtp_host) 

75 if self.creds_params.get('smtp_starttls', False) and self.creds_params.get('smtp_starttls') == 'True': 

76 if smtp.has_extn('STARTTLS'): 

77 self.logger.debug("Using tls.") 

78 try: 

79 smtp.starttls() 

80 except smtplib.SMTPException: 

81 self.logger.error("unable to start an encrypted session.") 

82 raise 

83 try: 

84 smtp.ehlo() 

85 except smtplib.SMTPException: 

86 self.logger.error("Helo failed after encrypting the session.") 

87 raise 

88 else: 

89 TefloNotifierError('The Start TLS is not available for the server.') 

90 

91 if self.creds_params.get('smtp_user', False) and self.creds_params.get('smtp_password', False): 

92 if smtp.has_extn('AUTH'): 

93 try: 

94 smtp.login(self.creds_params.get('smtp_user'), self.creds_params.get('smtp_password')) 

95 except smtplib.SMTPAuthenticationError: 

96 self.logger.error('Failed to authentication. Please check the username and/or password.') 

97 raise 

98 except smtplib.SMTPException: 

99 self.logger.error('Looks like no suitable authentication method was found.') 

100 raise 

101 else: 

102 raise TefloNotifierError('Authentication is not available for the server.') 

103 mail = self._build_msg() 

104 smtp.sendmail(from_addr=self.sender, to_addrs=self.receivers, msg=mail) 

105 

106 finally: 

107 smtp.quit()

109 def _build_msg(self): 

110 """Build the EmailMessage object""" 

111 

112 mail = MIMEMultipart() 

113 if self.cc: 

114 mail['cc'] = ','.join(self.cc) 

115 mail['to'] = ','.join(self.receivers) 

116 mail['from'] = self.sender 

117 mail['subject'] = self.subject 

118 mail["User-Agent"] = "teflo-email-notifier" 

119 

120 body = MIMEText(self.body) 

121 mail.attach(body) 

122 

123 if self.attachments: 

124 for a in self.attachments: 

125 part = MIMEBase('application', 'octet-stream') 

126 with open(a, 'rb') as fp: 

127 part.set_payload(fp.read()) 

128 encoders.encode_base64(part) 

129 part.add_header('Content-disposition', 'attachment', filename=os.path.basename(a)) 

130 mail.attach(part) 

131 

132 return mail.as_string()
