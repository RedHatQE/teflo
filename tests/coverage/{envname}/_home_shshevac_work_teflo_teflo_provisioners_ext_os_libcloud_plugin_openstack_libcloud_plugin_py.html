72 def authenticate(self): 

73 """Openstack authentication. 

74 

75 This method will create a driver object with libcloud provider class. 

76 Once driver object is created, it will verify the authentication with 

77 openstack endpoint is valid. If invalid, an exception will be raised. 

78 

79 Apache libcloud recommends to test authentication to test calling a 

80 method to perform some request against openstack. Checking the 

81 networks is quick to determine if authentication is good. FAQ: 

82 - http://libcloud.readthedocs.io/en/latest/faq.html 

83 """ 

84 # ignore SSL 

85 libcloud.security.VERIFY_SSL_CERT = False 

86 

87 # suppress insecure request warning messages 

88 urllib3.disable_warnings(urllib3.exceptions.InsecureRequestWarning) 

89 

90 credentials = self.provider_credentials 

91 

92 # determine region 

93 try: 

94 if not credentials['region']: 

95 # set default region if no region is defined 

96 credentials['region'] = 'regionOne' 

97 except KeyError: 

98 credentials['region'] = 'regionOne' 

99 

100 # determine domain 

101 try: 

102 if not credentials['domain_name']: 

103 # set the default domain if no domain is defined 

104 credentials['domain_name'] = 'default' 

105 except KeyError: 

106 credentials['domain_name'] = 'default' 

107 

108 # create libcloud driver object 

109 self._driver = get_driver(Provider.OPENSTACK)( 

110 credentials['username'], 

111 credentials['password'], 

112 ex_tenant_name=credentials['tenant_name'], 

113 ex_force_auth_url=credentials['auth_url'].split('/v')[0], 

114 ex_force_auth_version='3.x_password', 

115 ex_domain_name=credentials['domain_name'], 

116 ex_force_service_region=credentials['region'] 

117 ) 

118 # test authentication 

119 try: 

120 self._driver.ex_list_networks() 

121 except InvalidCredsError as ex: 

122 raise OpenstackProviderError( 

123 'Authentication failed: %s' % ex

183 """Get the libcloud image object based on the image provided. 

184 

185 This method will fetch all images using libcloud and return the

191 :return: Image object. 

192 :rtype: object

195 _images = self.images 

196 

197 # filter images

359 def create_node(self, name, image, size, network, key_pair): 

360 """Create node. 

361 

362 This method will create a new node in openstack. The node will be 

363 created based on the resource specifications provided. At any point if 

364 the creation gets an exception, it will wait and retry creating the 

365 node. If maximum attempts are reached, an exception will be raised. 

366 

367 :param name: Node name. 

368 :type name: str 

369 :param image: Image name. 

370 :type image: str 

371 :param size: Size name. 

372 :type size: str 

373 :param network: Network name. 

374 :type network: str 

375 :param key_pair: Key pair to inject into node. 

376 :type key_pair: str 

377 :return: Node object. 

378 :rtype: object 

379 """ 

380 attempt = 1 

381 

382 self.logger.debug('Creating node %s.' % name) 

383 self.logger.debug('Node details:\n * keypair=%s\n * image=%s\n ' 

384 '* flavor=%s\n * networks=%s' % 

385 (key_pair, image, size, network)) 

386 

387 # cache image object 

388 _image = self.image_lookup(image) 

389 

390 # cache size object 

391 _size = self.size_lookup(size) 

392 

393 # network object 

394 _network = self.network_lookup(network) 

395 

396 # create node 

397 while attempt <= MAX_ATTEMPTS: 

398 try: 

399 node = self.driver.create_node( 

400 name=name, 

401 image=_image, 

402 size=_size, 

403 networks=_network, 

404 ex_keyname=key_pair 

405 ) 

406 self.logger.info('Successfully booted node %s.' % name) 

407 return node 

408 except Exception as ex: 

409 self.logger.error(ex) 

410 wait_time = random.randint(10, MAX_WAIT_TIME) 

411 self.logger.info('Attempt %s of %s: retrying in %s seconds' % 

412 (attempt, MAX_ATTEMPTS, wait_time)) 

413 time.sleep(wait_time) 

414 attempt += 1 

415 finally: 

416 self.unset_driver() 

417 

418 # reach this point, maximum attempts to create node reached 

419 raise OpenstackProviderError( 

420 'Maximum attempts reached to boot node %s.' % name 

421 )

456 def _create(self, name, image, size, network, key_pair, fip): 

457 """Create. 

458 

459 This method will create a resource (node) in openstack. It will 

460 create the node, wait for the node to finish building and optionally 

461 attach a floating ip address. At any point if an exception is raised 

462 trying to wait for the node to finish building or a floating ip 

463 cannot be attached. It will clean up the resource (node). 

464 

465 :param name: Node name. 

466 :type name: str 

467 :param image: Image name. 

468 :type image: str 

469 :param size: Size name. 

470 :type size: str 

471 :param network: Network name. 

472 :type network: str 

473 :param key_pair: Key pair to inject into node. 

474 :type key_pair: str 

475 :param fip: Floating ip pool. 

476 :type fip: str 

477 :return: Node fip. id. 

478 :rtype: str(s) 

479 """ 

480 self.logger.info('Provisioning node %s.' % name) 

481 

482 # create node 

483 try: 

484 node = self.create_node(name, image, size, network, key_pair) 

485 except Exception: 

486 self.logger.error("Failed to create node %s " % name) 

487 raise 

488 

489 # wait for node to complete building 

490 try: 

491 self.wait_for_building_finish(node) 

492 except Exception as ex: 

493 self.logger.error(ex) 

494 self.logger.error('Node %s did not finish building.' % node.name) 

495 self.delete_node(node) 

496 raise OpenstackProviderError('Node did not finish building.') 

497 

498 # attach floating ip to node 

499 try: 

500 ip = self.attach_floating_ip(node, fip) 

501 except Exception as ex: 

502 self.logger.error(ex) 

503 self.logger.error('Failed to attach fip to node %s.' % node.name) 

504 self.delete_node(node) 

505 raise OpenstackProviderError('Failed to attach fip.') 

506 

507 self.logger.info('Successfully provisioned node %s.' % name) 

508 

509 # if no floating ip is assigned get updated node details and look for private_ip 

510 # TODO: This might need more logic if we support a use case for more than one network specified 

511 if ip is None: 

512 node = self.driver.ex_get_node_details(node.id) 

513 ip = node.private_ips[-1] 

514 return ip, node.id

545 def wait_for_building_finish(self, node): 

546 """Wait until a node is finished building. 

547 

548 :param node: node object 

549 """ 

550 self.logger.info('Wait for node %s to finish building.' % node.name) 

551 

552 status = 0 

553 attempt = 1 

554 while attempt <= 30: 

555 node = self.driver.ex_get_node_details(node.id) 

556 state = getattr(node, 'state') 

557 msg = '%s. VM %s, STATE=%s' % (attempt, node.name, state) 

558 

559 if state.lower() == 'error': 

560 self.logger.info(msg) 

561 self.logger.error('VM %s got an into an error state!' % 

562 node.name) 

563 break 

564 elif state.lower() == 'running': 

565 self.logger.info(msg) 

566 self.logger.info('VM %s successfully finished building!' % 

567 node.name) 

568 status = 1 

569 break 

570 else: 

571 self.logger.info('%s, rechecking in 20 seconds.', msg) 

572 time.sleep(20) 

573 

574 self.unset_driver() 

575 if status: 

576 self.logger.info('Node %s successfully finished building.' % 

577 node.name) 

578 else: 

579 raise OpenstackProviderError('Node was unable to build, %s' % 

580 node.name)

583 """Attach a floating ip address to a node. 

584 

585 This method will get the floating ip pool object, create a floating 

586 ip within that pool and attach to the node provided. 

587 

588 :param node: Node object. 

589 :type node: object 

590 :param fip: Floating ip pool. 

591 :type fip: str 

592 """

594 if not fip: 

595 self.logger.warning('Node %s does not require fip.' % node.name) 

596 return 

597 

598 self.logger.info('Attaching fip to node %s.' % node.name)

612 finally:

616 (_ip.ip_address, node.name))

649 def create(self): 

650 """Create a node in openstack. 

651 

652 This method will call the openstack provider create method. The 

653 provider class handles all interactions with the provider. 

654 """ 

655 self.logger.info('Provisioning machines from %s', self.__class__) 

656 

657 # ignore if count is given as provider parameter 

658 try: 

659 if self.provider_params.get('count', False): 

660 self.logger.warn('Count parameter is found for host %s ' 

661 'Count is not supported with openstack_libcloud as provisioner and will be ignored.' 

662 % getattr(self.asset, 'name')) 

663 except KeyError: 

664 pass 

665 

666 # determine hostname for the host 

667 hostname = self.provider_params.get('hostname', None) 

668 if not hostname: 

669 hostname = filter_host_name(getattr(self.asset, 'name')) + '_%s' % gen_random_str(5) 

670 

671 _ip, _id = self._create( 

672 hostname, 

673 self.provider_params.get('image'), 

674 self.provider_params.get('flavor'), 

675 self.provider_params.get('networks'), 

676 self.provider_params.get('keypair'), 

677 self.provider_params.get('floating_ip_pool', None) 

678 ) 

679 

680 return [dict(hostname=hostname, asset_id=_id, ip=_ip)]
