152 def __set_provider_attr__(self, parameters): 

153 """Configure the report provider attributes. 

154 

155 :param parameters: content which makes up the report resource 

156 :type parameters: dict 

157 :return: updated parameters 

158 :rtype: dict""" 

159 

160 # this is for backward compatibility, in case the provider key is used in the SDF 

161 self._provider = {} 

162 if parameters.get('provider', False): 

163 creds = parameters.get('provider').pop('credential', None) 

164 for p, v in parameters.pop('provider').items(): 

165 if p == 'name': 

166 self._provider = v 

167 continue 

168 setattr(self, p, v) 

169 self.has_provider = True 

170 else: 

171 # set no provider object 

172 creds = parameters.pop('credential', None) 

173 for p, v in parameters.items(): 

174 setattr(self, p, v) 

175 del self.provider 

176 self.has_provider = False 

177 

178 # lets set the credentials if any 

179 if creds: 

180 for item in self.config['CREDENTIALS']: 

181 if item['name'] == creds: 

182 self._credential = item 

183 break 

184 return parameters

280 def profile(self): 

281 """Builds a profile for the report resource. 

282 

283 :return: the report profile 

284 :rtype: OrderedDict 

285 """ 

286 filtered_attr = {k: v for k, v in vars(self).items() if not k.startswith('_') and k not in 

287 ['do_import', 'has_provider', 'all_hosts']} 

288 profile = OrderedDict() 

289 profile['name'] = self.name 

290 profile['description'] = self.description 

291 if not isinstance(self.importer_plugin, string_types): 

292 profile['importer'] = getattr( 

293 self.importer_plugin, '__plugin_name__') 

294 else: 

295 profile['importer'] = self.importer_plugin 

296 if hasattr(self, 'provider') and len(getattr(self, 'provider', {})) != 0: 

297 profile.update(OrderedDict(provider={})) 

298 profile.get('provider').update(name=self.provider) 

299 profile.get('provider').update(credential=self.credential.get('name')) 

300 profile.get('provider').update(filtered_attr) 

301 else: 

302 profile.update(credential=self.credential.get('name')) 

303 profile.update(filtered_attr) 

304 

305 # set the labels for report resource 

306 profile['labels'] = self.labels 

307 

308 # set the report's executes 

309 if all(isinstance(item, string_types) for item in self.executes): 

310 profile.update(executes=[execute for execute in self.executes]) 

311 else: 

312 profile.update(dict(executes=[execute.name for execute in self.executes])) 

313 

314 # set the report's import results 

315 profile.update({'import_results': self.import_results}) 

316 

317 return profile
