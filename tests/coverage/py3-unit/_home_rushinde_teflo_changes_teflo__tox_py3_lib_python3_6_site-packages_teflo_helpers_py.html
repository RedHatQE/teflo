71def get_core_tasks_classes(): 

72 """ 

73 Go through all modules within teflo.tasks package and return 

74 the list of all tasks classes within it. All tasks within the teflo.tasks 

75 module are considered valid task class to be added into the pipeline. 

76 :return: List of all valid tasks classes 

77 """ 

78 from .core import TefloTask 

79 from . import tasks

84 tasks_list = [] 

85 

86 # Run through each module within tasks and take the list of 

87 # classes that are subclass of TefloTask but not TefloTask itself. 

88 # When you import a class within a module, it becames a member of 

89 # that class 

90 for importer, modname, ispkg in pkgutil.iter_modules(tasks.__path__, prefix): 

91 if str(modname).endswith('.ext'):

93 clsmembers = inspect.getmembers(sys.modules[modname], inspect.isclass)

403def file_mgmt(operation, file_path, content=None, cfg_parser=None): 

404 """A generic function to manage files (read/write). 

405 

406 :param operation: File operation type to perform 

407 :type operation: str 

408 :param file_path: File name including path 

409 :type file_path: str 

410 :param content: Data to write to a file 

411 :type content: object 

412 :param cfg_parser: Config parser object (Only needed if the file being 

413 processed is a configuration file parser language) 

414 :type cfg_parser: bool 

415 :return: Data that was read from a file 

416 """ 

417 # to maintain the sequence in the results.yml file with ruamel 

418 yaml = YAML() 

419 yaml.default_flow_style = False 

420 yaml.representer.ignore_aliases = lambda *data: True 

421 yaml.Representer.add_representer(OrderedDict, yaml.Representer.represent_dict) 

422 

423 # Determine file extension 

424 file_ext = os.path.splitext(file_path)[-1] 

425 

426 if operation in ['r', 'read']: 

427 # Read 

428 if os.path.exists(file_path): 

429 if file_ext == ".json": 

430 # json 

431 with open(file_path) as f_raw: 

432 return json.load(f_raw) 

433 elif file_ext in ['.yaml', '.yml']: 

434 # yaml 

435 with open(file_path) as f_raw: 

436 return yaml.load(f_raw) 

437 else: 

438 # text 

439 with open(file_path) as f_raw: 

440 if cfg_parser is not None: 

441 # Config parser file 

442 return cfg_parser.readfp(f_raw) 

443 else: 

444 # lets check if it is json 

445 data = f_raw.read() 

446 try: 

447 return json.load(data) 

448 except Exception: 

449 # it wasn't json, lets try yaml 

450 try: 

451 return yaml.load(data) 

452 except Exception: 

453 # it wasn't yaml, lets just return pure string 

454 return data 

455 else: 

456 raise IOError("%s file not found!" % file_path) 

457 elif operation in ['w', 'write']: 

458 # Write 

459 mode = 'w+' if os.path.exists(file_path) else 'w' 

460 if file_ext == ".json": 

461 # json 

462 with open(file_path, mode) as f_raw: 

463 json.dump(content, f_raw, indent=4, sort_keys=True) 

464 elif file_ext in ['.yaml', '.yml']: 

465 # yaml 

466 with open(file_path, mode) as f_raw: 

467 yaml.dump(content, f_raw) 

468 else: 

469 # text 

470 with open(file_path, mode) as f_raw: 

471 if cfg_parser is not None: 

472 # Config parser file 

473 cfg_parser.write(f_raw) 

474 else: 

475 f_raw.write(content) 

476 elif operation in ['d', 'delete']: 

477 if os.path.exists(file_path): 

478 os.unlink(file_path) 

479 else: 

480 raise HelpersError("Unknown file operation: %s." % operation)

535def exec_local_cmd_pipe(cmd, logger, env_var=None): 

536 """Execute command locally, and pipe output in real time. 

537 

538 :param cmd: command to run 

539 :type cmd: str 

540 :param env_var: a dictionary of environmental variables to pass to the subprocess 

541 :type env_var: dictionary 

542 :param logger: logger object 

543 :type logger: object 

544 :return: tuple of rc and error (if there was an error) 

545 """ 

546 # updating passed env variables with os env variables

565 logger.info(output.strip()) 

566 rc = proc.poll() 

567 if rc != 0: 

568 error = proc.stderr.readline().decode('utf-8') 

569 return rc, error

624def filter_notifications_on_trigger(state, notify_list, passed_tasks, failed_tasks): 

625 """ 

626 Go through the notify_list and return notifications with which were skipped. 

627 If no notification with on_demand is found the original list is returned 

628 :return: List of notifications based on_demand 

629 """ 

630 if state == 'on_demand': 

631 return [res for res in notify_list if getattr(res, 'on_demand')] 

632 else: 

633 # filter out all on_demand notifications 

634 notify_list = [res for res in notify_list if not getattr(res, 'on_demand')] 

635 

636 if state == 'on_start': 

637 return [res for res in notify_list if getattr(res, 'on_start') and len(set(getattr(res, 'on_tasks', [])). 

638 intersection(passed_tasks)) != 0] 

639 elif state == 'on_complete': 

640 ocl = list() 

641 

642 # filter out all on_start notifications 

643 nl = [res for res in notify_list if not getattr(res, 'on_start')] 

644 

645 # not executing if on_success but there are failed tasks 

646 passed = [nt for nt in nl if getattr(nt, 'on_success') is True and getattr(nt, 'on_failure') is False] 

647 passed = [nt for nt in passed if len(failed_tasks) == 0 and len(set(getattr(nt, 'on_tasks')). 

648 intersection(passed_tasks)) != 0] 

649 ocl.extend(passed) 

650 

651 # not executing if on_failure but there are passed tasks 

652 failed = [nt for nt in nl if getattr(nt, 'on_failure') is True and getattr(nt, 'on_success') is False] 

653 failed = [nt for nt in failed if len(passed_tasks) == 0 and len(set(getattr(nt, 'on_tasks')). 

654 intersection(failed_tasks)) != 0] 

655 ocl.extend(failed) 

656 

657 # not executing if on_failure or on_success 

658 mixed = [nt for nt in nl if getattr(nt, 'on_success') is True and getattr(nt, 'on_failure') is True] 

659 mixed = [nt for nt in mixed if len(set(getattr(nt, 'on_tasks')). 

660 intersection(failed_tasks)) != 0 or len(set(getattr(nt, 'on_tasks')). 

661 intersection(passed_tasks)) != 0] 

662 ocl.extend(mixed) 

663 

664 return ocl

827def ssh_retry(obj): 

828 """ 

829 Decorator to check SSH Connection before method execution. 

830 Will perform 30 retries with sleep of 10 seconds 

831 between attempts 

832 """ 

833 MAX_ATTEMPTS = 30 

834 MAX_WAIT_TIME = 10 

835 

836 def check_access(*args, **kwargs): 

837 """ 

838 SSH Connection check and retries 

839 """ 

840 # Set flag and Inventory 

841 ssh_errs = False 

842 args[0].set_inventory() 

843 

844 # put everything into a list for rather than doing repetative if else statements 

845 # especially now that the inventory 'groups' property can be a string list of hosts 

846 # in the master inventory vs what is in the unique inventory 

847 host_groups = [kwargs['extra_vars']['hosts']] if kwargs['extra_vars']['hosts'].find(', ') == -1 \ 

848 else kwargs['extra_vars']['hosts'].split(', ') 

849 inv_groups = args[0].inventory.groups 

850 for host_group in host_groups: 

851 if is_host_localhost(host_group): 

852 # Run Playbook/Module if localhost; no need to check. 

853 result = obj(*args, **kwargs) 

854 return result 

855 if host_group not in inv_groups: 

856 raise HelpersError( 

857 'ERROR: Unexpected error - Group %s not found in inventory file!' % kwargs['extra_vars']['hosts'] 

858 ) 

859 

860 def can_connect(group): 

861 

862 sys_vars = group.vars 

863 server_ip = group.hosts[0].address 

864 LOG.info(server_ip) 

865 

866 # skip ssh connectivity check if server is localhost 

867 if is_host_localhost(server_ip): 

868 return False 

869 

870 server_user = sys_vars['ansible_user'] 

871 server_key_file = sys_vars['ansible_ssh_private_key_file'] 

872 server_ssh_port = 22 if 'ansible_port' not in sys_vars else sys_vars.get('ansible_port') 

873 

874 # Perform SSH checks 

875 attempt = 1 

876 while attempt <= MAX_ATTEMPTS: 

877 try: 

878 ssh = SSHClient() 

879 ssh.set_missing_host_key_policy(WarningPolicy()) 

880 

881 # Test ssh connection 

882 ssh.connect(server_ip, 

883 port=server_ssh_port, 

884 username=server_user, 

885 key_filename=server_key_file, 

886 timeout=5) 

887 LOG.debug("Server %s - IP: %s is reachable." % 

888 (group, server_ip)) 

889 ssh.close() 

890 break 

891 except (BadHostKeyException, AuthenticationException, 

892 SSHException, socket.error) as ex: 

893 attempt = attempt + 1 

894 LOG.error(ex) 

895 LOG.error("Server %s - IP: %s is unreachable." % (group, 

896 server_ip)) 

897 if attempt <= MAX_ATTEMPTS: 

898 LOG.info('Attempt %s of %s: retrying in %s seconds' % 

899 (attempt, MAX_ATTEMPTS, MAX_WAIT_TIME)) 

900 time.sleep(MAX_WAIT_TIME) 

901 

902 # Check Max SSH Retries performed 

903 if attempt > MAX_ATTEMPTS: 

904 LOG.error( 

905 'Max Retries exceeded. SSH ERROR - Resource unreachable - Server %s - IP: %s!' % 

906 (group, server_ip) 

907 ) 

908 return True 

909 return False 

910 

911 for host_group in host_groups: 

912 inv_group = inv_groups[host_group] 

913 # This is just here for backwards compat. In case I've missed any 

914 # corner case 

915 if hasattr(inv_group, 'child_groups') and inv_group.child_groups: 

916 LOG.debug('In the child group block') 

917 for group in inv_group.child_groups: 

918 ssh_errs = can_connect(group) 

919 else: 

920 # Most cases should be falling into this block, 

921 # based on teflo returning the actual host asset name once its 

922 # done with its fetch_assets logic 

923 ssh_errs = can_connect(inv_group) 

924 

925 # Check for SSH Errors 

926 if ssh_errs: 

927 raise HelpersError( 

928 'ERROR: Unable to establish ssh connection with resources!' 

929 ) 

930 

931 # Run Playbook/Module 

932 result = obj(*args, **kwargs) 

933 return result 

934 

935 return check_access

960class DataInjector(object): 

961 """Data injector class. 

962 

963 This class is primarily used for injecting data into strings which the 

964 data to be replaced needs to come from a host resource. It is helpful 

965 in the cases where orchestrate or execute tasks require additional data. 

966 i.e. ip address, metadata, authentication details, etc. 

967 

968 --- 

969 

970 How does this work? 

971 

972 You have a command in your definition file that needs the ip address of a 

973 host resource. 

974 

975 command: /usr/bin/foo --ip { host01.ip_address[0] } --args .. 

976 

977 The host01 is a resource defined in the provision section of teflo and 

978 the ip_address is an attribute of the host01. This class will evaluate 

979 the string and lookup the ip_address[0] from the host01 resource object 

980 and update the string with the correct information. This makes it helpful 

981 when orchestrate/execute tasks require data from the hosts itself. 

982 """ 

983 def __init__(self, hosts): 

984 """Constructor. 

985 

986 :param hosts: teflo host resources 

987 :type hosts: list 

988 """ 

989 self.hosts = hosts 

990 

991 # regular expression to search for in the string 

992 # data to be injected needs to be in the format of 

993 # { host01.metadata.k1 } 

994 self.regexp = r"\{(.*?)\}" 

995 

996 # regex to check jsonpath strings 

997 self.exclusion_chk_str = r"^range|^[|.|$|@]|[\w|']+:" 

998 

999 def host_exist(self, node): 

1000 """Determine if the host defined in the string formatted var is valid. 

1001 

1002 In the case no host is found, an exception is raised. 

1003 

1004 :param node: node name 

1005 :type node: str 

1006 :return: teflo host resource matching based on node input 

1007 :rtype: object 

1008 """ 

1009 for host in self.hosts: 

1010 if node == getattr(host, 'name'): 

1011 return host 

1012 raise TefloError('Node %s not found!' % node) 

1013 

1014 def inject(self, command): 

1015 """Main worker. 

1016 

1017 This method will perform the data injection. 

1018 

1019 :param command: command to inject data into 

1020 :type command: str 

1021 :return: updated command 

1022 :rtype: str 

1023 """ 

1024 

1025 variables = list(map(str.strip, re.findall(self.regexp, command))) 

1026 

1027 if not variables.__len__(): 

1028 return command 

1029 

1030 for variable in variables: 

1031 if re.match(self.exclusion_chk_str, variable): 

1032 LOG.debug("JSONPath format was identified in the command %s." % variable) 

1033 continue 

1034 else: 

1035 value = None 

1036 _vars = variable.split('.') 

1037 node = _vars.pop(0) 

1038 

1039 # verify variable has a valid host set 

1040 host = self.host_exist(node) 

1041 

1042 for index, item in enumerate(_vars): 

1043 try: 

1044 # is the item intended to be a position in a list, if so 

1045 # get the key and position 

1046 key = item.split('[')[0] 

1047 pos = int(item.split('[')[1].split(']')[0]) 

1048 

1049 if value: 

1050 # get the latest value from the dictionary 

1051 value = value[key][pos] 

1052 else: 

1053 # get latest value from host 

1054 if hasattr(host, key) and index <= 0: 

1055 value = getattr(host, key)[pos] 

1056 if isinstance(value, str): 

1057 break 

1058 

1059 # is the value a dict, if so keep going! 

1060 if isinstance(value, dict): 

1061 continue 

1062 except IndexError: 

1063 # item is not intended to be a position in a list 

1064 

1065 # check if the item is an attribute of the host 

1066 if hasattr(host, item) and index <= 0: 

1067 value = getattr(host, item) 

1068 

1069 if isinstance(value, str): 

1070 # we know the value has no further traversing to do 

1071 break 

1072 # value is either a list or dict, more traversing to do 

1073 continue 

1074 else: 

1075 if value is None: 

1076 raise AttributeError('%s not found in host %s!' % 

1077 (item, getattr(host, 'name'))) 

1078 

1079 # check if the item's value is a dict and update the value 

1080 # for further traversing to do 

1081 try: 

1082 if isinstance(value[item], dict): 

1083 value = value[item] 

1084 continue 

1085 except KeyError: 

1086 raise TefloError('%s not found in %s' % (item, value)) 

1087 

1088 # final check to get value no more traversing required 

1089 if value: 

1090 value = value[item] 

1091 except KeyError: 

1092 raise TefloError('Unable to locate item %s!' % item) 

1093 

1094 command = command.replace('{ %s }' % variable, value) 

1095 return command

1165def find_artifacts_on_disk(data_folder, report_name, art_location=[]): 

1166 """ 

1167 Used by the Artifact Importer to to search a list of paths in the results folder 

1168 to see if they exist. If the Execute collected artifacts, it will check the 

1169 the .results/artifacts/ specifically for the artifacts. 

1170 

1171 If the Execute did not collect artifacts, it will walk the .results 

1172 looking for the artifacts 

1173 

1174 :param data_folder: the results directory 

1175 :type data_folder: path as a string 

1176 :param path_list: The list of artifacts to look for 

1177 :type path_list: a list containing a string of paths 

1178 :param art_location_found: Whether the Executes object collected artifacts 

1179 :type art_location_found: Boolean 

1180 :return: a list of artifacts that were found to be imported. 

1181 """ 

1182 fnd_paths = list() 

1183 # build the regex query from the report_name pattern specified 

1184 regquery = build_artifact_regex_query(report_name) 

1185

1190 walked_list = walk_results_directory(data_folder, fnd_paths)

1295def validate_render_scenario(scenario, temp_data=None): 

1296 """ 

1297 This method takes the absolute path of the scenario descriptor file and returns back a list of 

1298 data streams of scenario(s) after doing the following checks: 

1299 (1) Checks there is no yaml.safe_load error for the provided scenario file 

1300 (2) Checks for include section present in the scenario file 

1301 (3) Checks the include section has valid scenario file path and it is not empty 

1302 (4) Checks there is no yaml.safe_load error for scenario file in the include section 

1303 :param scenario: scenario file path 

1304 :type scenario: str 

1305 :param temp_data: the file path to jinja template vars data or a json dictionary of vars data 

1306 :type temp_data: dict or str 

1307 :return: scenario data stream(s) 

1308 :rtype: list of data streams 

1309 """ 

1310 scenario_stream_list = list() 

1311 

1312 if temp_data: 

1313 if os.path.isfile(temp_data): 

1314 temp_data = file_mgmt('r', temp_data) 

1315 else: 

1316 temp_data = json.loads(temp_data) 

1317 # Updating th eos.environ to the variable data 

1318 temp_data.update(os.environ) 

1319 else: 

1320 # if temp_data isNone, sending only the os.environment 

1321 temp_data = os.environ 

1322 try: 

1323 data = yaml.safe_load(template_render(scenario, temp_data)) 

1324 # adding master scenario as the first scenario data stream 

1325 scenario_stream_list.append(template_render(scenario, temp_data)) 

1326 if 'include' in data.keys(): 

1327 include_item = data['include'] 

1328 include_template = list() 

1329 if include_item is not None: 

1330 for item in include_item: 

1331 if os.path.isfile(item): 

1332 item = os.path.abspath(item) 

1333 # check to verify the data in included scenario is valid 

1334 try: 

1335 yaml.safe_load(template_render(item, temp_data)) 

1336 include_template.append(template_render(item, temp_data)) 

1337 except yaml.YAMLError: 

1338 # raising Teflo error to differentiate the yaml issue is with included scenario 

1339 raise TefloError('Error loading updated included scenario data!') 

1340 else: 

1341 # raising HelperError if included file is invalid or included section is empty 

1342 raise HelpersError('Included File is invalid or Include section is empty .'

1351def ssh_key_file_generator(workspace, ssh_key_param): 

1352 """ 

1353 A method to find if the ssh key value in a provider 

1354 param is a path to a public or private key. If it 

1355 is a public key, just return the key back. If it 

1356 is a private key, generate the public key from it 

1357 and return that back. 

1358 

1359 Right now this is only really used by Linchpin 

1360 in the LinchpinResourceBuilder class for Beaker 

1361 resources because they expect a public key and the 

1362 various ssh key options in their resource defintions 

1363 

1364 :param workspace: the Teflo workspace keys directory 

1365 :type workspace: path as a string 

1366 :param ssh_key_param: the ssh_key from the provider param 

1367 :type ssh_key_param: value of the ssh_key param either a path or an actual key 

1368 :return: a path to a public key 

1369 """ 

1370 

1371 # setup absolute path for key 

1372 key = os.path.join(workspace, ssh_key_param) 

1373 

1374 # set permission of the key 

1375 try: 

1376 os.chmod(key, stat.S_IRUSR | stat.S_IWUSR) 

1377 except OSError as ex: 

1378 raise HelpersError( 

1379 'Error setting private key file permissions: %s' % ex 

1380 ) 

1381 

1382 # Lets assume it's a private key file and try to load it 

1383 # and create a public key for it 

1384 try: 

1385 rsa_key = RSAKey.from_private_key_file(key) 

1386 # generate public key from private 

1387 public_key = os.path.join( 

1388 workspace, ssh_key_param + ".pub" 

1389 ) 

1390 with open(public_key, 'w') as f: 

1391 f.write('%s %s\n' % (rsa_key.get_name(), rsa_key.get_base64())) 

1392 return public_key 

1393 except SSHException: 

1394 # Exception means the key file was invalid. 

1395 # Assume it's a public key and return it 

1396 return key

1497def create_individual_testrun_results(artifact_locations, config): 

1498 """this method creates a summary of total tests passed, failed, skipped for all the xml files found 

1499 as artifacts 

1500 :param artifact_locations: list of relative paths of artifacts where root dir is the key and artifact names are 

1501 values 

1502 :type artifact_locations: list 

1503 :param config: config parameter used by execute resource 

1504 :type config: dict 

1505 :return testruns: a dictionary of test results summary for individual xml files as well as aggregate of all xml 

1506 files found 

1507 :rtype testruns: dict 

1508 """ 

1509 fnd_paths = list() 

1510 individual_res = list() 

1511 # build the regex query to get only xml files 

1512 regquery = build_artifact_regex_query('*.xml') 

1513 # search the artifact location dictionary provided to search in the .results folder 

1514 fnd_paths.extend(search_artifact_location_dict(artifact_locations, '*.xml', config.get('RESULTS_FOLDER'), regquery)) 

1515 try: 

1516 for path in fnd_paths: 

1517 trun = dict() 

1518 trun['total_tests'] = 0 

1519 trun['failed_tests'] = 0 

1520 trun['skipped_tests'] = 0 

1521 trun['passed_tests'] = 0 

1522 tree = ET.parse(path) 

1523 root = tree.getroot() 

1524 temp = list() 

1525 # if root.tag is testsuites then collect all the element which are 'testsuite' 

1526 # if root.tag is testsuite then just collect that element 

1527 if root.tag == "testsuites": 

1528 temp.extend(root.findall('testsuite')) 

1529 elif root.tag == "testsuite": 

1530 temp.append(root) 

1531 if temp: 

1532 for test in temp: 

1533 trun['total_tests'] = trun['total_tests'] + len(test.findall('testcase')) 

1534 trun['failed_tests'] = trun['failed_tests'] + len([testcase.find('failure') 

1535 for testcase in test.findall('testcase') 

1536 if testcase.findall('failure')]) 

1537 trun['skipped_tests'] = trun['skipped_tests'] + len([testcase.find('skipped') 

1538 for testcase in test.findall('testcase') 

1539 if testcase.findall('skipped')]) 

1540 trun['passed_tests'] = trun['total_tests'] - trun['failed_tests'] - trun['skipped_tests'] 

1541 individual_res.append({os.path.basename(path): trun}) 

1542 else: 

1543 LOG.warning("The xml file %s does not have the correct format (no 'testsuite' or 'testsuites'" 

1544 " tags) to collect testrun results" % path) 

1545 continue 

1546 except ET.ParseError: 

1547 raise TefloError("The xml file %s is malformed " % path) 

1548 return individual_res

1552 """this method creates a aggregate summary of total tests passed, failed, skipped for all the xml files found 

1553 as artifacts 

1554 :param individual_results: list of individual test summaries of xml files found as artifacts
